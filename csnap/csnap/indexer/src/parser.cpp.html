16struct ParsingWork
17{
18 TranslationUnit* source = nullptr;
19 std::filesystem::path sourcefile;
20};
21
22static TranslationUnitParsingResult parse_translation_unit(libclang::Index& index, const ParsingWork& work)
23{
24 TranslationUnitParsingResult result;
25 result.source = work.source;
26
27 auto start = std::chrono::high_resolution_clock::now();
28 result.result = std::make_unique<libclang::TranslationUnit>(index.parseTranslationUnit(work.sourcefile.string(),
29 work.source->compile_options->includedirs,
30 CXTranslationUnit_DetailedPreprocessingRecord));
31 auto end = std::chrono::high_resolution_clock::now();
32 result.parsing_time = std::chrono::duration_cast<std::chrono::milliseconds>(end - start);
33
34 return result;
35}
36
37class ParseTranslationUnit : public Runnable
38{
39public:
40 libclang::Index& index;
41 ParsingWork work;
42 ParsingResultQueue& results;
43
44public:
45
46 ParseTranslationUnit(libclang::Index& idx, ParsingWork w, ParsingResultQueue& rqueue) :
47 index(idx),
48 work(std::move(w)),
49 results(rqueue)
50 {
51
52 }
53
54 void run() override
55 {
56 TranslationUnitParsingResult result = parse_translation_unit(index, work);
57 results.write(std::move(result));
58 }
59};

61/**
62 * \brief constructs a parser
63 * \param index the clang index that will be used to create the clang translation units
64 * \param flist the list of files that may be sent to the parser
65 */
66Parser::Parser(libclang::Index& index, const FileList& flist) :
67 m_index(index),
68 m_files(flist),
69 m_result_queue(std::make_unique<ParsingResultQueue>()),
70 m_threads(1)
71{
72
73}
74
75Parser::~Parser()
76{
77 if (!results().empty())
78 {
79 std::cout << "Warning: results() isn't empty in ~Parser()" << std::endl;
80 }
81}

83/**
84 * \brief sets the number of threads used for parsing
85 * \param n the number of threads
86 */
87void Parser::setThreadCount(size_t n)
88{
89 n = std::clamp(n, size_t(1), (size_t)std::thread::hardware_concurrency());
90 m_threads.setThreadCount(n);
91}
92
93/**
94 * \brief parse a translation unit asynchronously
95 *
96 * The source file associated with @a tu must be in the FileList passed
97 * to the constructor.
98 */
99void Parser::asyncParse(TranslationUnit* tu)
100{
101 if (!tu)
102 return;
103
104 std::filesystem::path p = m_files.get(FileId(tu->sourcefile_id))->path;
105
106 ParsingWork w;
107 w.source = tu;
108 w.sourcefile = p;
109
110 m_threads.run(new ParseTranslationUnit(m_index, std::move(w), *m_result_queue));
111}
112
113/**
114 * \brief returns whether all asynchronous parsing tasks have been completed
115 */
116bool Parser::done() const
117{
118 return m_threads.done();
119}
120
121/**
122 * \brief returns the parser results queue
123 */
124ParsingResultQueue& Parser::results()
125{
126 return *m_result_queue;
127}
