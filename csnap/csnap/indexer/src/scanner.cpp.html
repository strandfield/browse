17void save_to_db(TranslationUnitParsingResult& parsingResult, Snapshot& snapshot)
18{
19 auto path = std::filesystem::temp_directory_path() / (std::to_string(parsingResult.source->id.value()) + ".cxxtranslationunit");
20
21 parsingResult.result->saveTranslationUnit(path.string());
22
23 snapshot.addTranslationUnitSerializedAst(parsingResult.source, path);
24
25 std::filesystem::remove(path);
26}
27
28void process_indexing_result(IndexingResult& idxres, IndexingResultAggregator& aggregator)
29{
30 aggregator.reduce(idxres.references);
31
32 Snapshot& snapshot = aggregator.snapshot();
33
34 for (std::unique_ptr<File>& f : idxres.files)
35 {
36 snapshot.addFile(std::move(f));
37 }
38
39 idxres.files.clear();
40
41 snapshot.addIncludes(idxres.includes, idxres.source);
42
43 snapshot.addSymbols(idxres.symbols);
44
45 for (const std::pair<const SymbolId, std::vector<BaseClass>>& p : idxres.bases)
46 {
47 snapshot.addBases(p.first, p.second);
48 }
49
50 snapshot.addSymbolReferences(idxres.references);
51
52 snapshot.writePendingData();
53}

55/**
56 * \brief creates an empty snapshot
57 * \param p the path of the database
58 */
59void Scanner::initSnapshot(std::filesystem::path& p)
60{
61 m_snapshot = std::make_unique<Snapshot>(Snapshot::create(p));
62
63 m_snapshot->setProperty("csnap.version", csnap::versionstring());
64}

66/**
67 * \brief fills the snapshot by scanning a Visual Studio solution
68 * \param slnPath path to the sln file
69 *
70 * \warning initSnapshot() must be called before calling this function.
71 */
72void Scanner::scanSln(const std::filesystem::path& slnPath)
73{
74 openSln(slnPath, *m_snapshot);
75
76 m_snapshot->writePendingData();
77
78 libclang::LibClang clang;
79 libclang::Index index = clang.createIndex();
80
81 Parser parser{ index, m_snapshot->files() };
82 parser.setThreadCount(this->nb_parsing_threads);
83
84 for (TranslationUnit* tu : m_snapshot->translationUnits().all())
85 {
86 parser.asyncParse(tu);
87 }
88
89 // We have some time before parsing results become available,
90 // we use this time to save the file's content into database:
91 m_snapshot->addFilesContent();
92
93 Indexer indexer{ index, *m_snapshot };
94 IndexingResultAggregator aggregator{ *m_snapshot };
95
96 while (!parser.done() || !parser.results().empty())
97 {
98 TranslationUnitParsingResult pr{ parser.results().next() };
99
100 if (pr.result)
101 {
102 if (save_ast)
103 {
104 save_to_db(pr, *m_snapshot);
105 }
106
107 indexer.asyncIndex(std::move(pr));
108 }
109
110 while (!indexer.results().empty())
111 {
112 IndexingResult idxres{ indexer.results().next() };
113 process_indexing_result(idxres, aggregator);
114 }
115 }
116
117 while (!indexer.done() || !indexer.results().empty())
118 {
119 IndexingResult idxres{ indexer.results().next() };
120 process_indexing_result(idxres, aggregator);
121 }
122}
