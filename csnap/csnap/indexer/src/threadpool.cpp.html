7#include <algorithm>

9namespace csnap

11
12class ThreadPoolData
13{
14public:
15 SharedQueue<RunnablePtr> tasks;
16 size_t nb_active_tasks = 0;
17 std::mutex mutex;
18 std::condition_variable cv;
19};
20
21static void thread_pool_proc(ThreadPoolData& pooldata)
22{
23 for (;;)
24 {
25 RunnablePtr task{ pooldata.tasks.next() };
26
27 if (!task)
28 return;
29
30 task->run();
31
32 {
33 std::lock_guard lock{ pooldata.mutex };
34 --pooldata.nb_active_tasks;
35 }
36
37 pooldata.cv.notify_one();
38 }
39}
40
41ThreadPool::ThreadPool(size_t nb_threads) :
42 m_nb_active_threads(0),
43 m_data(std::make_unique<ThreadPoolData>())
44{
45 setThreadCount(nb_threads);
46}
47
48ThreadPool::~ThreadPool()
49{
50 {
51 std::lock_guard lock{ m_data->mutex };
52 // $TODO: check that there are no nullptr tasks in the queue #correctness
53 m_data->nb_active_tasks -= m_data->tasks.size();
54 m_data->tasks.clear();
55 }
56
57 for (size_t i(0); i < threadCount(); ++i)
58 {
59 m_data->tasks.write(nullptr);
60 }
61
62 for (std::thread& t : m_threads)
63 t.join();
64}
65
66void ThreadPool::run(Runnable* r)
67{
68 std::lock_guard lock{ m_data->mutex };
69 m_data->tasks.write(RunnablePtr(r));
70 m_data->nb_active_tasks += 1;
71}
72
73size_t ThreadPool::threadCount() const
74{
75 // $Warning: this is not m_threads.size()
76 return m_nb_active_threads;
77}
78
79void ThreadPool::setThreadCount(size_t n)
80{
81 cleanupThreads();
82
83 while (n < m_nb_active_threads)
84 {
85 m_data->tasks.write(nullptr);
86 --m_nb_active_threads;
87 }
88
89 while (m_nb_active_threads < n)
90 {
91 m_threads.push_back(std::thread(thread_pool_proc, std::ref(*m_data)));
92 ++m_nb_active_threads;
93 }
94}
95
96bool ThreadPool::done() const
97{
98 std::lock_guard lock{ m_data->mutex };
99 return m_data->nb_active_tasks == 0;
100}
101
102void ThreadPool::waitForDone()
103{
104 std::unique_lock lock{ m_data->mutex };
105
106 m_data->cv.wait(lock, [&]() {
107 return m_data->nb_active_tasks == 0;
108 });
109}
110
111void ThreadPool::cleanupThreads()
112{
113 auto it = std::remove_if(m_threads.begin(), m_threads.end(), [](const std::thread& t) {
114 return !t.joinable();
115 });
116
117 m_threads.erase(it, m_threads.end());
118}
