20static void listFiles(Snapshot& ss, const vcxproj::Solution& solution)
21{
22 std::vector<File> files;
23
24 for (const vcxproj::Project& p : solution.projects)
25 {
26 for (std::string f : p.includeList)
27 files.push_back(create_file(std::move(f)));
28
29 for (std::string f : p.compileList)
30 files.push_back(create_file(std::move(f)));
31 }
32
33 ss.addFiles(files);
34}
35
36static void listTranslationUnits(Snapshot& ss, const vcxproj::Solution& solution)
37{
38 for (const vcxproj::Project& p : solution.projects)
39 {
40 if (p.itemDefinitionGroupList.empty())
41 {
42 if (!p.compileList.empty())
43 {
44 std::cout << "Sln-Warning: project '" << p.name << "' empty itemDefinitionGroupListbut non-empty compileList" << std::endl;
45 }
46
47 continue;
48 }
49
50 program::CompileOptions params;
51
52 const vcxproj::ItemDefinitionGroup& idg = p.itemDefinitionGroupList.back();
53
54 std::vector<std::string> incdirs = vcxproj::split(idg.additionalIncludeDirectories, ';');
55
56 for (const std::string& incdir : incdirs)
57 params.includedirs.insert(incdir);
58
59 std::vector<std::string> defines = vcxproj::split(idg.preprocessorDefinitions, ';');
60
61 for (const std::string& def : defines)
62 {
63 size_t p = def.find('=');
64
65 if (p != std::string::npos)
66 {
67 std::string name = def.substr(0, p);
68 std::string value = def.substr(p + 1);
69 params.defines[name] = value;
70 }
71 else
72 {
73 params.defines[def] = "";
74 }
75 }
76
77 std::vector<FileId> list;
78 list.reserve(p.compileList.size());
79
80 for (std::string f : p.compileList)
81 {
82 File* file = ss.findFile(f);
83
84 if (!file)
85 {
86 // @TODO: log some error
87 continue;
88 }
89
90 list.push_back(file->id);
91 }
92
93 ss.addTranslationUnits(list, params);
94 }
95}

97/**
98 * \brief opens a Visual Studio solution
99 * \param path path to the sln file
100 * \param snapshot the snapshot in which the results are written
101 *
102 * This function lists the file and translation units in the various projects
103 * of the solution and adds them to the snapshot.
104 */
105void openSln(const std::filesystem::path& path, Snapshot& snapshot)
106{
107 vcxproj::Solution solution = vcxproj::load_solution(path);
108
109 snapshot.setProperty("sln.path", path.u8string());
110
111 listFiles(snapshot, solution);
112 listTranslationUnits(snapshot, solution);
113}
