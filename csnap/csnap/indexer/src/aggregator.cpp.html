33/**
34 * \brief removes all already references that are already known in the snapshot
35 * \param references a list of references
36 */
37void IndexingResultAggregator::reduce(std::vector<SymbolReference>& references)
38{
39 if (references.empty())
40 return;
41
42 auto comp = [](const SymbolReference& lhs, const SymbolReference& rhs) {
43 return std::make_tuple(lhs.file_id, lhs.line, lhs.col) <
44 std::make_tuple(rhs.file_id, rhs.line, rhs.col);
45 };
46
47 std::sort(references.begin(), references.end(), comp);
48
49 auto begin = references.begin();
50 auto end = begin;
51
52 do
53 {
54 int current_file_id = begin->file_id;
55 auto is_not_current_file = [&current_file_id](const SymbolReference& r) {
56 return r.file_id != current_file_id;
57 };
58 end = std::find_if(begin, references.end(), is_not_current_file);
59
60 size_t nb_refs = std::distance(begin, end);
61
62 auto it = m_files_data.find(FileId(current_file_id));
63
64 if (it == m_files_data.end())
65 {
66 m_files_data[FileId(current_file_id)].num_references = nb_refs;
67 }
68 else
69 {
70 if (it->second.num_references != nb_refs)
71 {
72 // Mismatch, we need to compare to see the difference.
73 // (this branch should be unlikely to happen)
74 std::vector<SymbolReference> refsinfile = snapshot().listReferencesInFile(FileId(current_file_id));
75
76 auto already_exists = [&refsinfile](const SymbolReference& r) {
77 return std::find(refsinfile.begin(), refsinfile.end(), r) != refsinfile.end();
78 };
79
80 auto newend = std::remove_if(begin, end, already_exists);
81 size_t remaining = std::distance(begin, newend);
82
83 if (newend != end)
84 {
85 //std::cout << "erasing " << std::distance(newend, end) << " already existing references" << std::endl;
86 end = references.erase(newend, end);
87 }
88
89 it->second.num_references += remaining;
90 }
91 else
92 {
93 // It's a match, we skip all the references
94 end = references.erase(begin, end);
95 }
96 }
97
98 begin = end;
99
100 } while (begin != references.end());
101
102}
