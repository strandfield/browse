23/*!
24 * \fn ~TranslationUnit()
25 */
26TranslationUnit::~TranslationUnit()
27{
28 if(translation_unit)
29 api->clang_disposeTranslationUnit(translation_unit);
30}
31
32/*!
33 * \fn TranslationUnit& operator=(TranslationUnit&& other)
34 */
35TranslationUnit& TranslationUnit::operator=(TranslationUnit&& other)
36{
37 if (this->api && this->translation_unit)
38 this->api->clang_disposeTranslationUnit(this->translation_unit);
39
40 this->api = other.api;
41 this->translation_unit = other.translation_unit;
42 other.translation_unit = nullptr;
43
44 return *(this);
45}
46
47/*!
48 * \fn Cursor getCursor() const
49 */
50Cursor TranslationUnit::getCursor() const
51{
52 CXCursor c = api->clang_getTranslationUnitCursor(this->translation_unit);
53 return Cursor{ *api, c };
54}
55
56/*!
57 * \fn Cursor getCursor(const SourceLocation& loc) const
58 */
59Cursor TranslationUnit::getCursor(const SourceLocation& loc) const
60{
61 CXCursor c = api->clang_getCursor(this->translation_unit, loc);
62 return Cursor{ *api, c };
63}
64
65/*!
66 * \fn CXSaveError saveTranslationUnit(const std::string& filename, unsigned options = CXSaveTranslationUnit_None) const
67 * \brief saves the translation unit
68 */
69CXSaveError TranslationUnit::saveTranslationUnit(const std::string& filename, unsigned options) const
70{
71 int r = api->clang_saveTranslationUnit(this->translation_unit, filename.c_str(), options);
72 return static_cast<CXSaveError>(r);
73}
74
75/*!
76 * \fn void suspendTranslationUnit()
77 * \brief suspends a translation unit
78 *
79 * Use \m reparseTranslationUnit() to resume the translation unit.
80 */
81void TranslationUnit::suspendTranslationUnit()
82{
83 api->clang_suspendTranslationUnit(*this);
84}
85
86/*!
87 * \fn CXErrorCode reparseTranslationUnit()
88 * \brief reparse a translation unit
89 */
90CXErrorCode TranslationUnit::reparseTranslationUnit()
91{
92 unsigned options = api->clang_defaultReparseOptions(*this);
93 int err = api->clang_reparseTranslationUnit(*this, 0, nullptr, options);
94 return static_cast<CXErrorCode>(err);
95}
96
97/*!
98 * \fn TokenSet tokenize(CXSourceRange range) const
99 */
100TokenSet TranslationUnit::tokenize(CXSourceRange range) const
101{
102 CXToken* tokens = nullptr;
103 unsigned int size = 0;
104 api->clang_tokenize(translation_unit, range, &tokens, &size);
105 return TokenSet{ *api, translation_unit, tokens, size };
106}

110 */
