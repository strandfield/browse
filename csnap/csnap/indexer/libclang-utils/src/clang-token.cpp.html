10std::string TokenSet::getSpelling() const
11{
12 if (size() == 0)
13 return std::string();
14
15 std::string result = at(0).getSpelling();
16
17 libclang::SourceRange range = at(0).getExtent();
18 libclang::SpellingLocation loc = range.getRangeEnd().getSpellingLocation();
19
20 for (size_t i = 1; i < size(); i++)
21 {
22 range = at(i).getExtent();
23 libclang::SpellingLocation tokloc = range.getRangeStart().getSpellingLocation();
24
25 if (tokloc.line != loc.line || tokloc.col != loc.col)
26 result.push_back(' ');
27
28 result += at(i).getSpelling();
29
30 loc = range.getRangeEnd().getSpellingLocation();
31 }
32
33 return result;
34}
