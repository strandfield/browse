21/*!
22 * \fn File getIncludedFile() const
23 * \brief returns the included file of a file inclusion cursor
24 */
25File Cursor::getIncludedFile() const
26{
27 return File(*api, api->clang_getIncludedFile(*this));
28}
29
30 /*!
31 * \fn size_t childCount() const
32 * \brief returns the number of children of this cursor
33 */
34size_t Cursor::childCount() const
35{
36 size_t counter = 0;
37
38 visitChildren([&counter](const Cursor&) {
39 ++counter;
40 });
41
42 return counter;
43}
44
45/*!
46 * \fn Cursor childAt(size_t index) const
47 * \brief access a child by index
48 */
49Cursor Cursor::childAt(size_t index) const
50{
51 Cursor result{ *api, this->cursor };
52
53 visitChildren([&result, &index](bool& stop_token, const Cursor& c) {
54 if (index == 0)
55 {
56 result = c;
57 stop_token = true;
58 }
59 --index;
60 });
61
62 return result;
63}
64
65/*!
66 * \fn std::vector<Cursor> children() const
67 * \brief returns the cursor's children
68 */
69std::vector<Cursor> Cursor::children() const
70{
71 std::vector<Cursor> result;
72
73 visitChildren([&](const Cursor& c) {
74 result.push_back(c);
75 });
76
77 return result;
78}
