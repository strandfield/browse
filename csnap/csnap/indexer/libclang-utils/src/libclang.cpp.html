18template<typename T>
19void resolve(dynlib::Library& libclang, const char* name, T& callback)
20{
21 callback = (T) libclang.resolve(name);
22
23 if (!callback)
24 throw LibClangError{ ("could not resolve libclang function : " + std::string(name)).c_str() };
25}
26
27static CXVersion parse_clang_version(std::string str)
28{
29 CXVersion result;
30
31 auto is_digit = [](char c) { return c >= '0' && c <= '9'; };
32
33 auto it = str.begin();
34
35 while (!is_digit(*it)) ++it;
36
37 auto major_begin = it;
38
39 while (is_digit(*it)) ++it;
40
41 result.Major = std::stoi(std::string(major_begin, it));
42
43 while (!is_digit(*it)) ++it;
44
45 auto minor_begin = it;
46
47 while (is_digit(*it)) ++it;
48
49 result.Minor = std::stoi(std::string(minor_begin, it));
50
51 while (!is_digit(*it)) ++it;
52
53 auto subminor_begin = it;
54
55 while (it != str.end() && is_digit(*it)) ++it;
56
57 result.Subminor = std::stoi(std::string(subminor_begin, it));
58
59
60 return result;
61}
