93void BasicIndexer::indexDeclaration(const CXIdxDeclInfo* info)
94{
95 std::cout << " declaration " << getName(info->entityInfo) << " (" << info->entityInfo->USR << ") ";
96
97 CXFile cxfile;
98 unsigned line, col;
99
100 libclangAPI().clang_indexLoc_getFileLocation(info->loc, nullptr, &cxfile, &line, &col, nullptr);
101
102 std::cout << "@" << libclangAPI().file(cxfile).getFileName() << ":" << line << ":" << col;
103
104 std::cout << std::endl;
105}
