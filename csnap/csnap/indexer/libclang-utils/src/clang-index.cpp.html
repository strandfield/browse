22 /*!
23 * \fn TranslationUnit createTranslationUnit(const std::string& astfile)
24 * \brief creates a translation unit from an ast file
25 *
26 * AST files can be produced with TranslationUnit::saveTranslationUnit().
27 */
28TranslationUnit Index::createTranslationUnit(const std::string& astfile)
29{
30 CXTranslationUnit tu = api.clang_createTranslationUnit(this->index, astfile.data());
31 return TranslationUnit{ api, tu };
32}
33
34/*!
35 * \fn TranslationUnit parseTranslationUnit(const std::string& file, const std::set<std::string>& includedirs, int options = 0)
36 */
37TranslationUnit Index::parseTranslationUnit(const std::string& file, const std::set<std::string>& includedirs, int options)
38{
39 const char* command_line_args[128] = { nullptr };
40 std::vector<std::string> argv{ "-x", "c++", "-Xclang", "-ast-dump", "-fsyntax-only" };
41 for (const std::string& f : includedirs)
42 {
43 argv.push_back("--include-directory");
44 argv.push_back(f);
45 }
46 for (int i(0); i < argv.size(); ++i)
47 command_line_args[i] = argv.at(i).data();
48
49 CXTranslationUnit tu = nullptr;
50
51 CXErrorCode error = api.clang_parseTranslationUnit2(this->index, file.data(), command_line_args, static_cast<int>(argv.size()), nullptr, 0, options, &tu);
52
53 if (error)
54 throw std::runtime_error{ "Could not parse translation unit" };
55
56 return TranslationUnit{ api, tu };
57}
