15/*!
16 * \fn CXResult findIncludesInFile(const Cursor& cursor, const File& file, Func&& callback)
17 */
18template<typename Func>
19CXResult findIncludesInFile(const TranslationUnit& tu, const File& file, Func&& callback)
20{
21 details::CursorAndRangeVisitor<Func> context{ *tu.api, callback };
22 CXCursorAndRangeVisitor data;
23 data.context = &context;
24 data.visit = &details::cursor_and_range_visit_proc<Func>;
25 return tu.api->clang_findIncludesInFile(tu, file, data);
26}
