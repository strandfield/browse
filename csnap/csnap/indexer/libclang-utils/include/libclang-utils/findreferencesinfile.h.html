5#ifndef LIBCLANGUTILS_FINDREFERENCESINFILE_H
6#define LIBCLANGUTILS_FINDREFERENCESINFILE_H

8#include "libclang-utils/cursorandrangevisitor.h"
9
10namespace libclang
11{
12
13/*!
14 * \fn CXResult findReferencesInFile(const Cursor& cursor, const File& file, Func&& callback)
15 */
16template<typename Func>
17CXResult findReferencesInFile(const Cursor& cursor, const File& file, Func&& callback)
18{
19 details::CursorAndRangeVisitor<Func> context{ *cursor.api, callback };
20 CXCursorAndRangeVisitor data;
21 data.context = &context;
22 data.visit = &details::cursor_and_range_visit_proc<Func>;
23 return cursor.api->clang_findReferencesInFile(cursor, file, data);
24}
25
26} // namespace libclang

28#endif // LIBCLANGUTILS_FINDREFERENCESINFILE_H
