455template<typename T>
456struct VisitorData
457{
458 LibClang& libclang;
459 T& functor;
460 bool should_break = false;
461};
462
463struct VisitorSelector1 {};
464struct VisitorSelector2 : VisitorSelector1 {};
465
466template<typename F>
467void visitor_invoker(F&& func, bool& stop_token, const Cursor& cursor, VisitorSelector1)
468{
469 func(cursor);
470}
471
472template<typename F, typename = decltype(std::declval<F>()(std::declval<bool&>(), std::declval<const Cursor&>()))>
473void visitor_invoker(F&& func, bool& stop_token, const Cursor& cursor, VisitorSelector2)
474{
475 func(stop_token, cursor);
476}
477
478template<typename T>
479CXChildVisitResult generic_visit_callback(CXCursor c, CXCursor p, CXClientData client_data)
480{
481 VisitorData<T>& data = *static_cast<VisitorData<T>*>(client_data);
482 visitor_invoker(data.functor, data.should_break, Cursor{ data.libclang, c }, VisitorSelector2{});
483 return data.should_break ? CXChildVisit_Break : CXChildVisit_Continue;
484}
