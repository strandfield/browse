202template<typename IndexerType>
203inline void IndexAction::indexTranslationUnit(const TranslationUnit& tu, IndexerType& indexer, int options)
204{
205 IndexerCallbacks callbacks;
206 callbacks.abortQuery = &details::IndexerCallbacks<IndexerType>::abortQuery;
207 callbacks.diagnostic = &details::IndexerCallbacks<IndexerType>::diagnostic;
208 callbacks.enteredMainFile = &details::IndexerCallbacks<IndexerType>::enteredMainFile;
209 callbacks.ppIncludedFile = &details::IndexerCallbacks<IndexerType>::ppIncludedFile;
210 callbacks.importedASTFile = &details::IndexerCallbacks<IndexerType>::importedASTFile;
211 callbacks.startedTranslationUnit = &details::IndexerCallbacks<IndexerType>::startedTranslationUnit;
212 callbacks.indexDeclaration = &details::IndexerCallbacks<IndexerType>::indexDeclaration;
213 callbacks.indexEntityReference = &details::IndexerCallbacks<IndexerType>::indexEntityReference;
214
215 api.clang_indexTranslationUnit(*this, &indexer, &callbacks, sizeof(IndexerCallbacks), options, tu);
216}
