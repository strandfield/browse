12namespace libclang
13{
14
15namespace details
16{
17
18template<typename T>
19struct CursorAndRangeVisitor
20{
21 LibClang& libclang;
22 T& functor;
23};
24
25template<typename T>
26CXVisitorResult cursor_and_range_visit_proc(void* contextptr, CXCursor cursor, CXSourceRange source_range)
27{
28 auto& context = *static_cast<CursorAndRangeVisitor<T>*>(contextptr);
29 context.functor(Cursor(context.libclang, cursor), SourceRange(context.libclang, source_range));
30 return CXVisit_Continue;
31}
