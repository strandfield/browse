22#if defined(OS_WINDOWS)
23static void replace_slashes(std::string& str)
24{
25 for (char& c : str)
26 {
27 if (c == '/')
28 c = '\\';
29 }
30}
31#endif
32
33struct LibraryImpl
34{
35 std::string libname;
36 bool loaded = false;
37 std::string error_string;
38
39#if defined(OS_WINDOWS)
40 HMODULE system_library = nullptr;
41#else
42 void* system_library = nullptr;
43#endif // defined(OS_WINDOWS)
44
45public:
46 LibraryImpl(std::string filename)
47 : libname(std::move(filename))
48 {
49#if defined(OS_WINDOWS)
50 replace_slashes(libname);
51#else
52 libname.append(".so");
53#endif
54 }
55};
56
57Library::Library(Library&& other)
58 : d(std::move(other.d))
59{
60}
61
62Library::~Library()
63{
64 if (!d)
65 return;
66
67#if defined(OS_WINDOWS)
68 if (d->loaded)
69 {
70 FreeLibrary(d->system_library);
71 d->system_library = nullptr;
72 }
73#else
74 if (d->loaded)
75 {
76 dlclose(d->system_library);
77 d->system_library = nullptr;
78 }
79#endif
80}
81
82Library::Library(const std::string& libname)
83 : d(new LibraryImpl(libname))
84{
85
86}
87
88bool Library::load()
89{
90 if (d->loaded)
91 return true;
92
93#if defined(OS_WINDOWS)
94 d->system_library = LoadLibrary(d->libname.c_str());
95 d->loaded = (d->system_library != nullptr);
96
97 if (!d->loaded)
98 {
99 DWORD error = GetLastError();
100 char buf[256];
101 FormatMessageA(FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_IGNORE_INSERTS,
102 NULL, error, MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),
103 buf, (sizeof(buf) / sizeof(char)), NULL);
104 d->error_string = buf;
105 }
106#else
107 d->system_library = dlopen(d->libname.c_str(), RTLD_LAZY);
108 d->loaded = (d->system_library != nullptr);
109
110 if(!d->loaded)
111 {
112 char* err = dlerror();
113
114 if(err)
115 {
116 d->error_string = err;
117 }
118 }
119#endif
120
121 return d->loaded;
122}
123
124bool Library::isLoaded() const
125{
126 return d->loaded;
127}
128
129const std::string& Library::errorString() const
130{
131 return d->error_string;
132}
133
134FunctionPointer Library::resolve(const char* sym)
135{
136 if (!load())
137 return nullptr;
138
139#if defined(OS_WINDOWS)
140 return (FunctionPointer)GetProcAddress(d->system_library, sym);
141#else
142 return (FunctionPointer)dlsym(d->system_library, sym);
143#endif
144}

146Library& Library::operator=(Library&& other)
147{
148 d = std::move(other.d);
149
150 return *this;
