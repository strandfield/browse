15static void write_file(const char* filename, const char* content)
16{
17 std::ofstream file{ filename };
18 file << content << std::endl;
19}
20
21static bool findLibclangParser()
22{
23 try
24 {
25 libclang::LibClang clang_parser;
26 std::cout << "libclang parser available, version = " << clang_parser.printableVersion() << std::endl;
27 return true;
28 }
29 catch (const libclang::LibClangError& err)
30 {
31 std::cout << "libclang error: " << err.what() << std::endl;
32 std::cout << "libclang parser tests will be skipped..." << std::endl;
33 }
34
35 return false;
36}
37
38static bool skipTest()
39{
40 static bool value = findLibclangParser();
41 return !value;
42}
43
44TEST_CASE("The parser is able to parse a function", "[libclang]")
45{
46 if (skipTest())
47 return;
48
49 write_file("test.cpp",
50 "const int foo(int& i, int j = 0) { return i+j; }");
51
52 libclang::LibClang libclang;
53 libclang::Index index = libclang.createIndex();
54
55 REQUIRE_NOTHROW(index.parseTranslationUnit("test.cpp", {}));
56}
57
58TEST_CASE("The parser is able to parse a simple function body", "[libclang]")
59{
60 if (skipTest())
61 return;
62
63 write_file("test.cpp",
64 "void foo(int n) noexcept { if(n > 0){} }");
65
66 libclang::LibClang libclang;
67 libclang::Index index = libclang.createIndex();
68
69 libclang::TranslationUnit tu = index.parseTranslationUnit("test.cpp", {});
70}
