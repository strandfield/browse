28/**
29 * \brief a thread-safe queue
30 *
31 * All public functions in this class are thread-safe.
32 * Please note, however, that the class was designed for a multiple-producers /
33 * single-consumer architecture; its API is therefore not that practical
34 * in a multiple-consumers scenario.
35 */
36template<typename T>
37class SharedQueue
38{
39public:
40
41 typedef T element_type;
42
43 static_assert(std::is_default_constructible<element_type>::value, "SharedQueue::element_type must be default-constructible");
44 static_assert(std::is_move_constructible<element_type>::value, "SharedQueue::element_type must be move-constructible");
45
46 SharedQueue() :
47 m_synchronization(std::make_unique<details::SharedQueueSynchronizationData>())
48 {
49
50 }
51
52 SharedQueue(const SharedQueue&) = delete;
53
54 /**
55 * \brief fetch and remove the next element from the queue
56 *
57 * If the queue is currently empty, the function waits until a new element
58 * is appended to the queue.
59 */
60 T next()
61 {
62 std::unique_lock<std::mutex> lock{ mutex() };
63
64 cv().wait(lock, [&]() {
65 return !container().empty();
66 });
67
68 T n{ std::move(container().front()) };
69 container().pop();
70 return n;
71 }
72
73 /**
74 * \brief wait for the next element
75 * \param d the maximum amount of time to wait
76 * \return true if an element is available, false otherwise
77 *
78 * If the queue isn't empty, this returns immediately.
79 */
80 bool waitForNext(std::chrono::milliseconds d)
81 {
82 std::unique_lock<std::mutex> lock{ mutex() };
83
84 auto pred = [&]() {
85 return !container().empty();
86 };
87
88 if (pred())
89 return true;
90
91 cv().wait_for(lock, d, pred);
92
93 return pred();
94 }
95
96 /**
97 * \brief appends an element to the queue
98 * \param val the element
99 *
100 * Writing an element may wake-up another thread waiting in next()
101 * or waitForNext().
102 */
103 void write(T val)
104 {
105 {
106 std::lock_guard<std::mutex> lock{ mutex() };
107 container().push(std::move(val));
108 }
109
110 cv().notify_one();
111 }
112
113 /**
114 * \brief returns whether the queue is empty
115 *
116 * \warning If more than one thread is consuming elements from the queue,
117 * the caller should not assume that calling next() will return immediately
118 * after empty() returned true: indeed, another thread may have taken the
119 * last element of the queue in the meantime.
120 */
121 bool empty() const
122 {
123 std::lock_guard<std::mutex> lock{ mutex() };
124 return container().empty();
125 }
126
127 /**
128 * \brief returns the number of elements in the queue
129 *
130 * \sa empty().
131 */
132 size_t size() const
133 {
134 std::lock_guard<std::mutex> lock{ mutex() };
135 return container().size();
136 }
137
138 /**
139 * \brief removes all elements from the queue
140 */
141 void clear()
142 {
143 std::lock_guard<std::mutex> lock{ mutex() };
144 container() = {};
145 }
146
147protected:
148
149 std::mutex& mutex() const
150 {
151 return m_synchronization->mutex;
152 }
153
154 std::condition_variable& cv() const
155 {
156 return m_synchronization->cv;
157 }
158
159 std::queue<T>& container()
160 {
161 return m_queue;
162 }
163
164 const std::queue<T>& container() const
165 {
166 return m_queue;
167 }
168
169 SharedQueue<T> operator=(const SharedQueue<T>&) = delete;
170
171private:
172 std::queue<T> m_queue;
173 std::unique_ptr<details::SharedQueueSynchronizationData> m_synchronization;
174};
