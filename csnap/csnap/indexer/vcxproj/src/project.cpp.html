14inline bool streq(const char* a, const char* b)
15{
16 return std::strcmp(a, b) == 0;
17}
18
19inline bool elementis(const tinyxml2::XMLElement& e, const char* name)
20{
21 return streq(e.Name(), name);
22}

24class XmlChildElementIterator
25{
26public:
27 const tinyxml2::XMLElement* child = nullptr;
28
29public:
30 explicit XmlChildElementIterator(const tinyxml2::XMLElement* c)
31 : child(c)
32 {
33
34 }
35
36 XmlChildElementIterator(const XmlChildElementIterator&) = default;
37
38 const tinyxml2::XMLElement& operator*() const
39 {
40 return *child;
41 }
42
43 XmlChildElementIterator& operator++()
44 {
45 child = child->NextSiblingElement();
46 return *this;
47 }
48
49 XmlChildElementIterator operator++(int)
50 {
51 XmlChildElementIterator copy{ *this };
52 ++(*this);
53 return copy;
54 }
55
56 XmlChildElementIterator& operator=(const XmlChildElementIterator&) = default;
57};
58
59bool operator!=(const XmlChildElementIterator& lhs, const XmlChildElementIterator& rhs)
60{
61 return lhs.child != rhs.child;
62}

92template<typename T>
93T from_xml(const tinyxml2::XMLElement*);
94
95template<>
96vcxproj::ProjectConfiguration from_xml(const tinyxml2::XMLElement* xml)
97{
98 vcxproj::ProjectConfiguration ret;
99
100 const char* a = xml->Attribute("Include");
101 if (a)
102 ret.name = a;
103
104 for (const tinyxml2::XMLElement& e : children(*xml))
105 {
106 if (!e.GetText())
107 continue;
108
109 if (elementis(e, "Configuration"))
110 {
111 ret.configuration = e.GetText();
112 }
113 else if (elementis(e, "Platform"))
114 {
115 ret.platformStr = e.GetText();
116
117 if (ret.platformStr == "Win32")
118 ret.platform = vcxproj::ProjectConfiguration::Win32;
119 else if (ret.platformStr == "x64")
120 ret.platform = vcxproj::ProjectConfiguration::x64;
121 else
122 ret.platform = vcxproj::ProjectConfiguration::Unknown;
123 }
124 }
125
126 return ret;
127}

129template<>
130vcxproj::ItemDefinitionGroup from_xml(const tinyxml2::XMLElement* xml)
131{
132 vcxproj::ItemDefinitionGroup ret;
133
134 const char* condAttr = xml->Attribute("Condition");
135
136 if (condAttr)
137 ret.condition = condAttr;
138
139 for (const tinyxml2::XMLElement& clcompile : children(*xml))
140 {
141 if (!elementis(clcompile, "ClCompile"))
142 continue;
143
144 for (const tinyxml2::XMLElement& e : children(clcompile))
145 {
146 if (!e.GetText())
147 continue;
148
149 if (elementis(e, "PreprocessorDefinitions"))
150 {
151 ret.preprocessorDefinitions = e.GetText();
152 }
153 else if (elementis(e, "AdditionalIncludeDirectories"))
154 {
155 ret.additionalIncludeDirectories += e.GetText();
156 }
157 else if (elementis(e, "LanguageStandard"))
158 {
159 if (streq(e.GetText(), "stdcpp14"))
160 ret.cppstd = vcxproj::CppStandard::Cpp14;
161 else if (streq(e.GetText(), "stdcpp17"))
162 ret.cppstd = vcxproj::CppStandard::Cpp17;
163 else if (streq(e.GetText(), "stdcpp20"))
164 ret.cppstd = vcxproj::CppStandard::Cpp20;
165 else if (streq(e.GetText(), "stdcpplatest"))
166 ret.cppstd = vcxproj::CppStandard::CppLatest;
167 }
168 }
169 }
170
171 return ret;
172}

174class ProjectParser
175{
176public:
177 vcxproj::Project* project;
178 std::map<std::string, std::string>& variables;
179
180public:
181
182 ProjectParser(vcxproj::Project* p, std::map<std::string, std::string>& vars)
183 : project(p),
184 variables(vars)
185 {
186
187 }
188
189 void operator()(const tinyxml2::XMLElement* root)
190 {
191 for (const tinyxml2::XMLElement& node : children(*root))
192 {
193 if (elementis(node, "ItemGroup"))
194 {
195 const char* labelAttribute = node.Attribute("Label");
196
197 if (labelAttribute && streq(labelAttribute, "ProjectConfigurations"))
198 {
199 for (const tinyxml2::XMLElement& cfg : children(node))
200 {
201 if (elementis(cfg, "ProjectConfiguration"))
202 {
203 auto pconf = from_xml<vcxproj::ProjectConfiguration>(&cfg);
204
205 if (pconf.platform != vcxproj::ProjectConfiguration::Unknown)
206 project->projectConfigurationList.push_back(std::move(pconf));
207 }
208 }
209 }
210 else
211 {
212 for (const tinyxml2::XMLElement& e : children(node))
213 {
214 if (elementis(e, "ClCompile"))
215 {
216 const char* include = e.Attribute("Include");
217
218 if (include)
219 project->compileList.emplace_back(include);
220 }
221 else if (elementis(e, "ClInclude"))
222 {
223 const char* include = e.Attribute("Include");
224
225 if (include)
226 project->includeList.emplace_back(include);
227 }
228 }
229 }
230 }
231 else if (elementis(node, "ItemDefinitionGroup"))
232 {
233 project->itemDefinitionGroupList.push_back(from_xml<vcxproj::ItemDefinitionGroup>(&node));
234 }
235 }
236 }
237};

242bool evaluate(std::string& s, const std::map<std::string, std::string>& variables, bool readenv)
243{
244 size_t start = 0;
245
246 while ((start = s.find("$(", start)) != std::string::npos)
247 {
248 size_t end = s.find(')', start);
249
250 if (end == std::string::npos)
251 return false;
252
253 std::string var = s.substr(start + 2, end - (start + 2));
254
255 auto it = variables.find(var);
256
257 if (it != variables.end())
258 {
259 s.replace(start, end - start + 1, it->second);
260 }
261 else
262 {
263 if (!readenv)
264 return false;
265
266 const char* env = std::getenv(var.c_str());
267
268 if (!env)
269 return false;
270
271 s.replace(start, end + 1 - start, std::string(env));
272 }
273 }
274
275 return s.find("$(") == std::string::npos;
276}

278void load_project(Project* project, const std::filesystem::path& filepath, std::map<std::string, std::string>& variables)
279{
280 std::string fp = filepath.u8string();
281
282 ProjectParser parser{ project, variables };
283
284 tinyxml2::XMLDocument doc;
285
286 if (doc.LoadFile(fp.c_str()) != tinyxml2::XML_SUCCESS)
287 return;
288
289 const tinyxml2::XMLElement* rootnode = doc.FirstChildElement();
290
291 if (rootnode == nullptr)
292 return;
293
294 parser(rootnode);
295}
