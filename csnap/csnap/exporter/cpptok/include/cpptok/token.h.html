235/*!
236 * \class Token
237 * \brief represents a token
238 *
239 * Each token has a \c{type()} and a \c{text()}.
240 *
241 * Note that the token's text is stored as a string_view.
242 * This makes tokens cheap to copy but requires that the string
243 * that was used to create the token outlives the token.
244 */
245
246class Token
247{
248private:
249 TokenType type_;
250 string_view str_;
251
252public:
253 Token();
254 Token(const Token & ) = default;
255 ~Token() = default;
256
257 Token(TokenType t, string_view str);
258
259 bool isValid() const;
260
261 TokenType type() const;
262 string_view text() const;
263
264 std::string to_string() const;
265
266 bool isPunctuator() const;
267 bool isOperator() const;
268 bool isIdentifier() const;
269 bool isKeyword() const;
270 bool isLiteral() const;
271 bool isComment() const;
272
273 bool operator==(const Token& other) const { return type_ == other.type_ && other.str_ == str_; }
274 bool operator!=(const Token & other) const { return !operator==(other); }
275 bool operator==(TokenType tok) const { return type_ == tok; }
276 bool operator!=(TokenType tok) const { return !operator==(tok); }
277
278 Token & operator=(const Token&) = default;
279};
