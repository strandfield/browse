44/*!
45 * \fn void tokenize(const char* str, size_t len)
46 * \param the string to tokenize
47 * \param the length of the string
48 */
49void Tokenizer::tokenize(const char* str, size_t len)
50{
51 m_chars = str;
52 m_len = len;
53 m_pos = 0;
54 m_start = 0;
55
56 if (state == State::LongComment)
57 readMultiLineComment();
58
59 while (!atEnd())
60 read();
61}

76void Tokenizer::read()
77{
78 consumeDiscardable();
79
80 if (atEnd())
81 return;
82
83 m_start = pos();
84 size_t p = m_start;
85
86 char c = readChar();
87 CharacterType ct = ctype(c);
88
89 switch (ct)
90 {
91 case Digit:
92 return readNumericLiteral();
93 case DoubleQuote:
94 return readStringLiteral();
95 case SingleQuote:
96 return readCharLiteral();
97 case Letter:
98 case Underscore:
99 return readIdentifier();
100 case LeftPar:
101 return write(TokenType::LeftPar);
102 case RightPar:
103 return write(TokenType::RightPar);
104 case LeftBrace:
105 return write(TokenType::LeftBrace);
106 case RightBrace:
107 return write(TokenType::RightBrace);
108 case LeftBracket:
109 return write(TokenType::LeftBracket);
110 case RightBracket:
111 return write(TokenType::RightBracket);
112 case Semicolon:
113 return write(TokenType::Semicolon);
114 case Colon:
115 return readColonOrColonColon();
116 case QuestionMark:
117 return write(TokenType::QuestionMark);
118 case Comma:
119 return write(TokenType::Comma);
120 case Dot:
121 return write(TokenType::Dot);
122 case Punctuator:
123 return readFromPunctuator(c);
124 default:
125 return write(TokenType::Invalid);
126 }
127}

323void Tokenizer::readNumericLiteral()
324{
325 if (atEnd()) {
326 if (charAt(m_start) == '0')
327 return write(TokenType::OctalLiteral);
328 else
329 return write(TokenType::IntegerLiteral);
330 }
331
332 char c = peekChar();
333
334 // Reading binary, octal or hexadecimal number
335 // eg. : 0b00110111
336 // 018
337 // 0xACDBE
338 if (charAt(m_start) == '0' && c != '.')
339 {
340 if (c == 'x') // hexadecimal
341 return readHexa();
342 else if (c == 'b') // binary
343 return readBinary();
344 else if (Tokenizer::isDigit(c))// octal
345 return readOctal();
346 else // it is zero
347 return write(TokenType::OctalLiteral);
348 }
349
350 return readDecimal();
351}
352
353void Tokenizer::readHexa()
354{
355 const char x = readChar();
356 assert(x == 'x');
357
358 if (atEnd()) // input ends with '0x' -> error
359 return write(TokenType::Invalid);
360
361 while (!atEnd() && Tokenizer::isHexa(peekChar()))
362 readChar();
363
364 if (pos() - m_start == 2) // e.g. 0x+
365 return write(TokenType::Invalid);
366
367 return write(TokenType::HexadecimalLiteral);
368}
369
370void Tokenizer::readOctal()
371{
372 while (!atEnd() && Tokenizer::isOctal(peekChar()))
373 readChar();
374
375 return write(TokenType::OctalLiteral);
376}

378void Tokenizer::readBinary()
379{
380 const char b = readChar();
381 assert(b == 'b');
382
383 if (atEnd()) // input ends with '0b' -> error
384 return write(TokenType::Invalid);
385
386 while (!atEnd() && Tokenizer::isBinary(peekChar()))
387 readChar();
388
389 return write(TokenType::BinaryLiteral);
390}

392void Tokenizer::readDecimal()
393{
394 // Reading decimal numbers
395 // eg. : 25
396 // 3.14
397 // 3.14f
398 // 100e100
399 // 6.02e23
400 // 6.67e-11
401
402 while (!atEnd() && Tokenizer::isDigit(peekChar()))
403 readChar();
404
405 if (atEnd())
406 return write(TokenType::IntegerLiteral);
407
408 bool is_decimal = false;
409
410 if (peekChar() == '.')
411 {
412 readChar();
413 is_decimal = true;
414
415 while (!atEnd() && Tokenizer::isDigit(peekChar()))
416 readChar();
417
418 if (atEnd())
419 return write(TokenType::DecimalLiteral);
420 }
421
422 if (peekChar() == 'e')
423 {
424 readChar();
425 is_decimal = true;
426
427 if (atEnd())
428 return write(TokenType::Invalid);
429
430 if (peekChar() == '+' || peekChar() == '-')
431 {
432 readChar();
433 if (atEnd())
434 return write(TokenType::Invalid);
435 }
436
437 while (!atEnd() && Tokenizer::isDigit(peekChar()))
438 readChar();
439
440 if (atEnd())
441 return write(TokenType::DecimalLiteral);
442 }
443
444
445 if (peekChar() == 'f') // eg. 125.f
446 {
447 readChar();
448 is_decimal = true;
449 }
450 else
451 {
452 if (tryReadLiteralSuffix())
453 return write(TokenType::UserDefinedLiteral);
454 }
455
456 return write(is_decimal ? TokenType::DecimalLiteral : TokenType::IntegerLiteral);
457}

459bool Tokenizer::tryReadLiteralSuffix()
460{
461 auto cpos = pos();
462
463 if (!this->atEnd() && (Tokenizer::isLetter(peekChar()) || peekChar() == '_'))
464 readChar();
465 else
466 return false;
467
468 while (!this->atEnd() && (Tokenizer::isLetter(peekChar()) || Tokenizer::isDigit(peekChar()) || peekChar() == '_'))
469 readChar();
470
471 const bool read = (cpos != pos());
472 return read;
473}

475void Tokenizer::readPreprocessor()
476{
477 consumeDiscardable();
478
479 if (atEnd() || !isIdentifier(peekChar()))
480 return write(TokenType::Invalid);
481
482 while (!atEnd() && isIdentifierOrDigit(peekChar()))
483 readChar();
484
485 write(TokenType::Preproc);
486
487 if (this->output.back().text() == "#include")
488 {
489 consumeDiscardable();
490 m_start = pos();
491
492 if (atEnd() || (peekChar() != '<' && peekChar() != '"'))
493 return;
494
495 char c = readChar();
496 c = c == '<' ? '>' : '"';
497
498 while (!atEnd() && peekChar() != c)
499 readChar();
500
501 if (atEnd())
502 return write(TokenType::Invalid);
503
504 readChar();
505
506 return write(TokenType::Include);
507 }
508}
509
510void Tokenizer::readIdentifier()
511{
512 while (!this->atEnd() && (Tokenizer::isLetter(peekChar()) || Tokenizer::isDigit(peekChar()) || peekChar() == '_'))
513 readChar();
514
515 return write(identifierType(m_start, pos()));
516}

676void Tokenizer::readStringLiteral()
677{
678 while (!atEnd() && peekChar() != '"')
679 {
680 if (peekChar() == '\\')
681 {
682 readChar();
683 if (!atEnd())
684 readChar();
685 }
686 else if (peekChar() == '\n')
687 {
688 return write(TokenType::Invalid);
689 }
690 else
691 {
692 readChar();
693 }
694 }
695
696 if(atEnd())
697 return write(TokenType::Invalid);
698
699 assert(peekChar() == '"');
700 readChar();
701
702 if (tryReadLiteralSuffix())
703 return write(TokenType::UserDefinedLiteral);
704
705 return write(TokenType::StringLiteral);
706}
707
708void Tokenizer::readCharLiteral()
709{
710 if(atEnd())
711 return write(TokenType::Invalid);
712
713 readChar();
714
715 if (atEnd())
716 return write(TokenType::Invalid);
717
718 if(ctype(readChar()) != SingleQuote)
719 return write(TokenType::Invalid);
720
721 return write(TokenType::StringLiteral);
722}

724void Tokenizer::readFromPunctuator(char p)
725{
726 if (p == '/')
727 {
728 if (atEnd())
729 return write(TokenType::Div);
730 if (peekChar() == '/')
731 return readSingleLineComment();
732 else if (peekChar() == '*')
733 return readMultiLineComment();
734 else
735 return readOperator();
736 }
737 else if (p == '#')
738 {
739 return readPreprocessor();
740 }
741
742 return readOperator();
743}

746void Tokenizer::readColonOrColonColon()
747{
748 if (atEnd())
749 return write(TokenType::Colon);
750
751 if (peekChar() == ':')
752 {
753 readChar();
754 return write(TokenType::ScopeResolution);
755 }
756
757 return write(TokenType::Colon);
758}

840void Tokenizer::readOperator()
841{
842 TokenType op = getOperator(m_start, pos());
843
844 if (op == TokenType::Invalid)
845 return write(TokenType::Invalid);
846
847 while (!atEnd())
848 {
849 size_t p = pos();
850 readChar();
851 TokenType candidate = getOperator(m_start, pos());
852
853 if (candidate == TokenType::Invalid)
854 {
855 --m_pos;
856 break;
857 }
858 else
859 {
860 op = candidate;
861 }
862 }
863
864 return write(op);
865}

867void Tokenizer::readSingleLineComment()
868{
869 readChar(); // reads the second '/'
870
871 while (!atEnd() && peekChar() != '\n')
872 readChar();
873
874 return write(TokenType::SingleLineComment);
875}

877void Tokenizer::createLongComment()
878{
879 state = State::LongComment;
880 return write(TokenType::MultiLineComment);

882
883void Tokenizer::readMultiLineComment()
884{
885 if(state == State::Default)
886 readChar(); // reads the '*' after opening '/'
887
888 do {
889 while (!atEnd() && peekChar() != '*')
890 readChar();
891
892 if (atEnd())
893 return createLongComment();
894
895 assert(peekChar() == '*');
896 readChar(); // reads the '*'
897
898 if (atEnd())
