10TEST_CASE("Tokenize keywords", "[cpptok]")
11{
12 cpptok::Tokenizer lexer;
13 lexer.tokenize(" auto bool break case catch char class const ");
14 lexer.tokenize("constexpr continue default delete do double else ");
15 lexer.tokenize(" enum explicit export false final float for friend ");
16 lexer.tokenize(" if import inline int mutable namespace noexcept operator ");
17 lexer.tokenize(" override private protected public return static struct ");
18 lexer.tokenize(" template this throw true try typedef typeid typename ");
19 lexer.tokenize(" using virtual void while ");
20 lexer.tokenize(" goto decltype const_cast static_cast dynamic_cast reinterpret_cast ");
21 lexer.tokenize(" static_assert extern unsigned long");
22
23 REQUIRE(lexer.output.size() == 60);
24
25 for (const cpptok::Token& tok : lexer.output)
26 {
27 REQUIRE(tok.isKeyword());
28 }
29}

44TEST_CASE("Tokenize simple decl", "[cpptok]")
45{
46 cpptok::Tokenizer lexer;
47 lexer.tokenize(" int n = 5; ");
48
49 REQUIRE(lexer.output.size() == 5);
50
51 REQUIRE(lexer.output[0].text() == "int");
52 REQUIRE(lexer.output[0].type() == cpptok::TokenType::Int);
53
54 REQUIRE(lexer.output[1].text() == "n");
55 REQUIRE(lexer.output[1].type() == cpptok::TokenType::UserDefinedName);
56
57 REQUIRE(lexer.output[2].text() == "=");
58 REQUIRE(lexer.output[2].type() == cpptok::TokenType::Eq);
59
60 REQUIRE(lexer.output[3].text() == "5");
61 REQUIRE(lexer.output[3].type() == cpptok::TokenType::IntegerLiteral);
62
63 REQUIRE(lexer.output[4].text() == ";");
64 REQUIRE(lexer.output[4].type() == cpptok::TokenType::Semicolon);
65}

67TEST_CASE("Tokenize multiline comment", "[cpptok]")
68{
69 cpptok::Tokenizer lexer;
70 lexer.tokenize(" foo() /* bar ");
71
72 REQUIRE(lexer.output.size() == 4);
73 REQUIRE(lexer.state == cpptok::Tokenizer::LongComment);
74
75 lexer.tokenize(" baz */ toast ");
76
77 REQUIRE(lexer.output.size() == 6);
78 REQUIRE(lexer.state == cpptok::Tokenizer::Default);
79
80 REQUIRE(lexer.output[3].text() == "/* bar ");
81 REQUIRE(lexer.output[4].text() == " baz */");
82}

84TEST_CASE("Tokenize preprocessor directive", "[cpptok]")
85{
86 cpptok::Tokenizer lexer;
87 lexer.tokenize("#define FOO");
88
89 REQUIRE(lexer.output.size() == 2);
90 REQUIRE(lexer.output[0].type() == cpptok::TokenType::Preproc);
91 REQUIRE(lexer.output[0].text() == "#define");
92 REQUIRE(lexer.output[1].text() == "FOO");
93
94 lexer.output.clear();
95
96 lexer.tokenize("#include <vector>");
97
98 REQUIRE(lexer.output.size() == 2);
99 REQUIRE(lexer.output[0].text() == "#include");
100 REQUIRE(lexer.output[1].type() == cpptok::TokenType::Include);
101 REQUIRE(lexer.output[1].text() == "<vector>");
102}
