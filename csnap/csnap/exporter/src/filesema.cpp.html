12inline static bool is_decl_or_def(const SymbolReference& r)
13{
14 return r.flags & (SymbolReference::Declaration | SymbolReference::Definition);
15}
16
17static bool swap_if_needed_and_discard_first(SymbolReference& first, SymbolReference& second, const std::map<SymbolId, std::shared_ptr<Symbol>>& symbols)
18{
19 std::shared_ptr<Symbol> first_symbol = find_symbol(symbols, SymbolId(first.symbol_id));
20 std::shared_ptr<Symbol> second_symbol = find_symbol(symbols, SymbolId(second.symbol_id));
21
22 if (!first_symbol || !second_symbol)
23 return false;
24
25 // The following if-elseif could probably be written in a more elegant way...
26 if (first_symbol->kind == Whatsit::CXXConstructor && second_symbol->kind == Whatsit::CXXClass)
27 {
28 if (is_decl_or_def(first))
29 {
30 std::swap(first, second);
31 }
32
33 return true;
34 }
35 else if (first_symbol->kind == Whatsit::CXXClass && second_symbol->kind == Whatsit::CXXConstructor)
36 {
37 if (!is_decl_or_def(second))
38 {
39 std::swap(first, second);
40 }
41
42 return true;
43 }
44
45 return false;
46}

48/**
49 * \brief simplify references to the class and its constructors at the same location
50 * \param refs the sorted vector of references within a file to process
51 * \param symbols the map of all symbols referenced in 'refs'
52 *
53 * To declare a constructor, one cannot avoid referencing the class name.
54 * This leads to both a reference to the class and to the constructor being located at
55 * the same position within a file.
56 *
57 * There are other situations where both a class and one of its constructor are
58 * referenced at the same location (e.g., when the ctor is implicitely called with a
59 * member of the class).
60 * void f(MyClass a); f(MyClass::MyEnumValue);
61 *
62 * When exporting a snapshot as HTML, one of the reference must be removed (depending on the context);
63 * which is what this function does.
64 * When the constructor is declared or defined, the class reference must be removed; otherwise
65 * the constructor reference must be removed.
66 */
67void simplify_ctor_and_class_references(std::vector<SymbolReference>& refs, const std::map<SymbolId, std::shared_ptr<Symbol>>& symbols)
68{
69 auto read = refs.begin();
70 auto write = read;
71
72 while (read != refs.end())
73 {
74 auto it = std::adjacent_find(read, refs.end(), [](const SymbolReference& lhs, const SymbolReference& rhs) {
75 return lhs.line == rhs.line && lhs.col == rhs.col;
76 });
77
78 if (it != refs.end())
79 {
80 if (swap_if_needed_and_discard_first(*it, *std::next(it), symbols))
81 {
82 std::copy(read, it, write);
83 write += std::distance(read, it);
84 ++it;
85 }
86 else
87 {
88 ++it;
89 std::copy(read, it, write);
90 write += std::distance(read, it);
91 }
92 }
93 else
94 {
95 std::copy(read, refs.end(), write);
96 write += std::distance(read, it);
97 }
98
99 read = it;
100 }
101
102 if (write != refs.end())
103 {
104 refs.erase(write, refs.end());
105 }
106}
