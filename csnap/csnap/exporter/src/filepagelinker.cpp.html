12/**
13 * \brief constructs a page linker
14 * \param url the url of the html page being written
15 */
16FilePageLinker::FilePageLinker(PageURL url) :
17 m_url(std::move(url))
18{
19
20}
21
22/**
23 * \brief returns the url of the current page
24 *
25 * Thhis is the url passed to the constructor unless it was
26 * subsequently modified using setCurrentPageUrl().
27 */
28const PageURL& FilePageLinker::currentPageUrl() const
29{
30 return m_url;
31}
32
33void FilePageLinker::setCurrentPageUrl(PageURL url)
34{
35 m_url = std::move(url);
36}
37
38/**
39 * \brief returns a pointer to the path resolver
40 */
41PathResolver* FilePageLinker::pathResolver() const
42{
43 return m_path_resolver;
44}
45
46/**
47 * \brief sets the path resolver used to link to other pages
48 */
49void FilePageLinker::setPathResolver(PathResolver& resolver)
50{
51 m_path_resolver = &resolver;
52}
53
54/**
55 * \brief computes a relative link to the html page for a file
56 * \param file the file to link to
57 *
58 * A valid PathResolver must have been specified using setPathResolver()
59 * before calling this function.
60 */
61std::string FilePageLinker::linkTo(const File& file) const
62{
63 assert(pathResolver());
64
65 std::filesystem::path p = pathResolver()->filePath(file);
66 return m_url.linkTo(p);
67}
68
69/**
70 * \brief computes a relative link to the html page for a file
71 * \param file the file to link to
72 * \param line the line to target
73 *
74 * \overload std::string FilePageLinker::linkTo(const File& file) const
75 */
76std::string FilePageLinker::linkTo(const File& file, int line) const
77{
78 return linkTo(file) + "#L" + std::to_string(line);
79}
80
81/**
82 * \brief returns whether this class is ready to produce links to other pages
83 */
84FilePageLinker::operator bool() const
85{
86 return m_path_resolver != nullptr;
87}
