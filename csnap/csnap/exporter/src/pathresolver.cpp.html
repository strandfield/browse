20std::string PathResolver::relpath(const std::string& source, const std::string& target)
21{
22 if (target == source)
23 {
24 size_t n = target.rfind('/');
25
26 if (n == std::string::npos)
27 return target;
28 else
29 return target.substr(n + 1);
30 }
31
32 size_t minlen = std::min(source.size(), target.size());
33 size_t i = 0;
34
35 while (i < minlen && target.at(i) == source.at(i)) ++i;
36
37 size_t nbdir = std::count(source.begin() + i, source.end(), '/');
38
39 std::string r;
40 r.reserve(target.size() - i + nbdir * 3);
41
42 while (nbdir--)
43 {
44 r.append("../");
45 }
46
47 r.append(target.begin() + i, target.end());
48
49 return r;
50}

52/**
53 * \brief returns a url for a given file

55 *
56 * The default implementation simply returns the path of the file.
