14/**
15 * \brief builds the table
16 * \param defs a list of all symbol definitions
17 *
18 * This function will sort @a defs and create a fast (O(1)) lookup table.
19 */
20void DefinitionTable::build(std::vector<SymbolReference> defs)
21{
22 m_definitions = std::move(defs);
23 m_table.clear();
24
25 if (m_definitions.empty())
26 return;
27
28 std::sort(m_definitions.begin(), m_definitions.end(),
29 [](const SymbolReference& a, const SymbolReference& b) -> bool {
30 return a.symbol_id < b.symbol_id;
31 });
32
33 int max_symbol_id = m_definitions.back().symbol_id;
34
35 m_table.resize(static_cast<size_t>(max_symbol_id) + 1, std::numeric_limits<size_t>::max());
36
37 int current_symbol_id = m_definitions.front().symbol_id;
38 size_t current_start_offset = 0;
39
40 for (size_t i(0); i < m_definitions.size(); ++i)
41 {
42 if (m_definitions.at(i).symbol_id != current_symbol_id)
43 {
44 auto symoffset = static_cast<size_t>(current_symbol_id);
45 m_table[symoffset] = current_start_offset;
46
47 current_symbol_id = m_definitions.at(i).symbol_id;
48 current_start_offset = i;
49 }
50 }
51}

53/**
54 * \brief returns whether a symbol is defined only once
55 * \param id the id of the symbol
56 * \param def output pointer where the definition is written
57 *
58 * If @a def is not nullptr, and there is only a single definition associated
59 * with the symbol (i.e., this function returns true); the definition is
60 * written in @a def.
61 */
62bool DefinitionTable::hasUniqueDefinition(SymbolId id, SymbolReference* def) const
63{
64 auto symoffset = static_cast<size_t>(id.value());
65
66 if (m_table.size() <= symoffset || m_table[symoffset] == std::numeric_limits<size_t>::max())
67 return false;
68
69 size_t firstdef = m_table[symoffset];
70
71 if (firstdef + 1 == m_definitions.size() || m_definitions[firstdef + 1].symbol_id != id.value())
72 {
73 if (def)
74 *def = m_definitions[firstdef];
75
76 return true;
77 }
78 else
79 {
80 return false;
81 }
82}
