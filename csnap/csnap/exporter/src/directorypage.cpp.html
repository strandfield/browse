16struct DirectoryContent
17{
18 std::vector<std::pair<File*, std::filesystem::path>> files;
19 std::set<std::filesystem::path> folders;
20};
21
22static DirectoryContent filter_directory_content(const std::map<File*, std::filesystem::path>& allpages, std::filesystem::path wd)
23{
24 std::string genwd = wd.generic_string();
25
26 DirectoryContent result;
27
28 for (const auto& p : allpages)
29 {
30 const std::string& path = p.second.generic_string();
31
32 if (path.length() <= genwd.length() || std::strncmp(genwd.c_str(), path.c_str(), genwd.size()) != 0) // not in folder
33 continue;
34
35 if (std::count(path.begin() + genwd.size(), path.end(), '/') == 1) // in the folder
36 {
37 result.files.push_back(std::make_pair(p.first, p.second));
38 }
39 else // in a nested folder
40 {
41 std::filesystem::path folderpath = p.second.parent_path().string().substr(genwd.size() + (!genwd.empty() ? 1 : 0));
42 result.folders.insert(folderpath);
43 }
44 }
45
46 return result;
47}
48
49static void filter_folders(std::set<std::filesystem::path>& paths)
50{
51 using Path = std::filesystem::path;
52
53 auto comp = [](const Path& lhs, const Path& rhs) {
54 const std::string& a = lhs.generic_string();
55 const std::string& b = lhs.generic_string();
56 return a.size() < b.size()
57 && b.at(a.size()) == '/'
58 && std::strncmp(a.c_str(), b.c_str(), a.size()) == 0;
59 };
60
61 auto it = std::adjacent_find(paths.begin(), paths.end(), comp);
62
63 while (it != paths.end())
64 {
65 it = paths.erase(it);
66 it = std::adjacent_find(it, paths.end(), comp);
67 }
68}

70static void sort(DirectoryContent& dircontent)
71{
72 using P = std::pair<File*, std::filesystem::path>;
73
74 std::sort(dircontent.files.begin(), dircontent.files.end(), [](const P& lhs, const P& rhs) {
75 return lhs.second < rhs.second;
76 });
77}

80static std::string get_filename(const std::string& path)
81{
82 size_t pos = path.rfind('/');
83 return pos == std::string::npos ? path : path.substr(pos + 1);
84}

86DirectoryPageGenerator::DirectoryPageGenerator(HtmlPage& p, const std::map<File*, std::filesystem::path>& allpages, std::filesystem::path wd) :
87 page(p),
88 all_pages(allpages),
89 working_dir(std::move(wd))
90{
91
92}
93
94void DirectoryPageGenerator::writePage()
95{
96 page.xml.write("<!DOCTYPE html>\n");
97
98 html::start(page);
99 {
100 html::head(page);
101 {
102 html::title(page);
103 page << working_dir.generic_string();
104 html::endtitle(page);
105 }
106 html::endhead(page);
107
108 html::body(page);
109 {
110 writeBody();
111 }
112 html::endbody(page);
113 }
114 html::end(page);
115}
116
117void DirectoryPageGenerator::writeBody()
118{
119 html::h1(page);
120 page << working_dir.generic_string();
121 html::endh1(page);
122
123 DirectoryContent content = filter_directory_content(all_pages, working_dir);
124 filter_folders(content.folders);
125 sort(content);
126
127 html::table(page);
128 {
129 html::tbody(page);
130 {
131 for (const std::filesystem::path& p : content.folders)
132 {
133 std::string foldername = p.string();
134 std::string href = foldername + "/index.html";
135
136 html::tr(page);
137 {
138 html::td(page);
139 {
140 html::a(page);
141 html::attr(page, "href", href);
142 page << foldername;
143 html::enda(page);
144 }
145 html::endtd(page);
146 }
147 html::endtr(page);
148 }
149
150 for (const auto& p : content.files)
151 {
152 std::string filename = get_filename(p.first->path);
153 std::string href = std::filesystem::relative(p.second, working_dir).string();
154
155 html::tr(page);
156 {
157 html::td(page);
158 {
159 html::a(page);
160 html::attr(page, "href", href);
161 page << filename;
162 html::enda(page);
163 }
164 html::endtd(page);
165 }
166 html::endtr(page);
167 }
168 }
169 html::endtbody(page);
170 }
171 html::endtable(page);
172}
