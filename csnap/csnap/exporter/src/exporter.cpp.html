27void export_html(Snapshot& snapshot, File& file, const std::filesystem::path& outputdir, const std::filesystem::path& outputpath, const DefinitionTable& defs, PathResolver& pathresolver)
28{
29 std::shared_ptr<FileContent> fc = snapshot.getFileContent(file.id);
30
31 if (!fc)
32 return;
33
34 FileSema sema;
35 sema.file = &file;
36 sema.references = snapshot.listReferencesInFile(file.id);
37 sema.includes = snapshot.listIncludesInFile(file.id);
38 sema.reverse_includes = snapshot.findWhereFileIsIncluded(file.id);
39
40 remove_implicit_references(sema.references);
41
42 FileBrowserGenerator::SymbolMap symbols;
43
44 {
45 std::set<SymbolId> ids;
46
47 for (const SymbolReference& ref : sema.references)
48 {
49 ids.insert(SymbolId(ref.symbol_id));
50
51 if (ref.parent_symbol_id.valid())
52 {
53 ids.insert(ref.parent_symbol_id);
54 }
55 }
56
57 snapshot.loadSymbols(ids, symbols);
58 }
59
60 simplify_ctor_and_class_references(sema.references, symbols);
61
62 std::stringstream outstrstream;
63 XmlWriter xml{ outstrstream };
64
65 HtmlPage page{ outputpath, xml };
66 page.links().setPathResolver(pathresolver);
67
68 FileBrowserGenerator generator{ page, *fc, std::move(sema), snapshot.files(), symbols, defs };
69 generator.generatePage();
70
71 write_file(outputdir / outputpath, outstrstream.str());
72}

74static std::set<std::filesystem::path> list_directories(const std::map<File*, std::filesystem::path>& paths)
75{
76 std::set<std::filesystem::path> result;
77
78 for (const auto& p : paths)
79 {
80 std::filesystem::path parent_path = p.second.parent_path();
81
82 if (parent_path != std::filesystem::path())
83 result.insert(parent_path);
84 }
85
86 return result;
87}
88
89static std::set<std::filesystem::path> list_directories_recursive(const std::map<File*, std::filesystem::path>& paths)
90{
91 std::set<std::filesystem::path> result = list_directories(paths);
92
93 for (auto it = result.begin(); it != result.end(); )
94 {
95 std::filesystem::path parent_path = it->parent_path();
96
97 if (parent_path != std::filesystem::path())
98 {
99 auto [newit, inserted] = result.insert(parent_path);
100
101 if (inserted)
102 it = newit;
103 else
104 ++it;
105 }
106 else
107 {
108 ++it;
109 }
110 }
111
112 return result;
113}

115static void export_symbol(Snapshot& snapshot, const Symbol& symbol, const std::filesystem::path& outputdir, const std::filesystem::path& outputpath, PathResolver& pathresolver)
116{
117 std::stringstream outstrstream;
118 XmlWriter xml{ outstrstream };
119
120 HtmlPage page{ outputpath, xml };
121 SymbolPageGenerator pagegen{ page, snapshot, symbol };
122
123 pagegen.setPathResolver(pathresolver);
124
125 pagegen.writePage();
126
127 write_file(outputdir / outputpath, outstrstream.str());
128}

130class SnapshotExporterHtmlPathResolver : public PathResolver
131{
132public:
133 std::string rootdir;
134
135public:
136 explicit SnapshotExporterHtmlPathResolver(std::string root = {}) :
137 rootdir(std::move(root))
138 {
139
140 }
141
142 std::filesystem::path filePath(const File& file) const override
143 {
144 static const std::string extension = ".html";
145
146 if (!rootdir.empty() && file.path.find(rootdir) == 0)
147 {
148 std::string path = file.path.substr(rootdir.size() + 1);
149 return std::filesystem::path(path + extension);
150 }
151 else
152 {
153 std::filesystem::path p{ file.path + extension };
154 return p.relative_path();
155 }
156 }
157};

174/**
175 * \brief runs the exporting process
176 */
177void SnapshotExporter::run()
178{
179 if (rootpath == "%auto%")
180 detectRootPath();
181
182 export_resources_html_assets(outputdir);
183
184 std::map<File*, std::filesystem::path> paths = writeFilePages();
185
186 writeDirectoryPages(paths);
187
188 writeSymbolPages();
189}

191void SnapshotExporter::detectRootPath()
192{
193 std::vector<File*> files = snapshot.files().all();
194
195 if (files.empty())
196 return;
197
198 rootpath = files.front()->path;
199
200 auto common_substr_len = [](const std::string& a, const std::string& b) -> size_t {
201 size_t n = std::min(a.size(), b.size());
202 size_t i = 0;
203 while (i < n && a.at(i) == b.at(i)) ++i;
204 return i;
205 };
206
207 for (File* f : files)
208 {
209 size_t n = common_substr_len(rootpath, f->path);
210 rootpath.erase(rootpath.begin() + n, rootpath.end());
211 }
212
213 if (!rootpath.empty())
214 {
215 rootpath.pop_back();
216
217 if (rootpath.size() == 2 && rootpath.back() == ':')
218 rootpath = "";
219
220 rootpath = std::filesystem::path(rootpath).parent_path().generic_string();
221 }
222
223 std::cout << "exporter root path detected: " << rootpath << std::endl;
224}

226std::map<File*, std::filesystem::path> SnapshotExporter::writeFilePages()
227{
228 DefinitionTable defs;
229 defs.build(select_symboldefinition(snapshot.database()));
230
231 SnapshotExporterHtmlPathResolver pathresolver{ rootpath };
232
233 std::vector<File*> files = snapshot.files().all();
234
235 std::map<File*, std::filesystem::path> result;
236
237 for (size_t i(0); i < files.size(); ++i)
238 {
239 File* f = files.at(i);
240 std::cout << "[" << (i + 1) << "/" << files.size() << "] " << f->path << std::endl;
241 std::filesystem::path savepath = pathresolver.filePath(*f);
242 result[f] = savepath;
243 export_html(snapshot, *f, outputdir, savepath, defs, pathresolver);
244 }
245
246 return result;
247}

249void SnapshotExporter::writeDirectoryPages(const std::map<File*, std::filesystem::path>& paths)
250{
251 std::cout << "Directory pages:" << std::endl;
252
253 std::set<std::filesystem::path> directories = list_directories_recursive(paths);
254
255 // $note: ugly hack to get a homepage
256 directories.insert(std::filesystem::path());
257
258 for (const std::filesystem::path& dirpath : directories)
259 {
260 std::cout << dirpath.generic_string() << std::endl;
261
262 std::stringstream outstrstream;
263 XmlWriter xml{ outstrstream };
264
265 HtmlPage page{ dirpath / "index.html", xml };
266
267 DirectoryPageGenerator gen{ page, paths, dirpath };
268 gen.writePage();
269
270 write_file(outputdir / page.url().path(), outstrstream.str());
271 }
272}

274void SnapshotExporter::writeSymbolPages()
275{
276 SymbolEnumerator symenumerator{ snapshot };
277
278 while (symenumerator.next())
279 {
280 const Symbol& symbol = symenumerator.symbol;
281
282 std::cout << symbol.display_name << std::endl;
283
284 SnapshotExporterHtmlPathResolver pathresolver{ rootpath };
285 std::string outputpath = "symbols/" + SourceHighlighter::symbol_symref(symbol) + ".html";
286 export_symbol(snapshot, symbol, outputdir, outputpath, pathresolver);
287 }
