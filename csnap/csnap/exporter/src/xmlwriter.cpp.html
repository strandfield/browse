9/**
10 * \brief constructs a xml writer on an output stream
11 * \param output the output stream
12 *
13 * The output stream must be derived from std::ostream.
14 * This makes this class suitable for both writing directly to a file
15 * or in memory in a string buffer.
16 */
17XmlWriter::XmlWriter(OutputStream& output)
18 : m_output(output)
19{
20
21}
22
23/**
24 * \brief returns the writer's output stream
25 */
26XmlWriter::OutputStream& XmlWriter::stream() const
27{
28 return m_output;
29}

31/**
32 * \brief write bytes to the output stream
33 *
34 * \warning Unlike writeCharacters(), this function does not escape
35 * special XML characters like '<'.
36 */
37void XmlWriter::write(const std::string& txt)
38{
39 stream() << txt;
40}

42/**
43 * \brief starts a new xml element
44 * \param qname the qualified named of the element
45 */
46void XmlWriter::writeStartElement(const std::string& qname)
47{
48 if (withinElement() && !current().attributes_closed)
49 sealAttributes();
50
51 m_output << "<" << qname;
52
53 Element e;
54 e.name = qname;
55
56 m_elements.push_back(std::move(e));
57}
58
59/**
60 * \brief closes the current element
61 *
62 * Depending on whether the element has inner text or nested elements, this will
63 * either close the element with "/>" or with "</tagname>".
64 */
65void XmlWriter::writeEndElement()
66{
67 if (!m_elements.back().attributes_closed)
68 {
69 m_output << "/>";
70 }
71 else
72 {
73 m_output << "</" << m_elements.back().name << ">";
74 }
75
76 m_elements.pop_back();
77}
78
79/**
80 * \brief add an attribute to the current element
81 * \param name the attribute's name
82 * \param value the attribute's value
83 *
84 * \warning You must write all attributes of an element immediately after it was opened
85 * with writeStartElement().
86 * Specifically, all calls to writeAttribute() must come before calls to writeCharacters()
87 * or writeStartElement() (which will open a new element).
88 */
89void XmlWriter::writeAttribute(const std::string& name, const std::string& value)
90{
91 assert(!current().attributes_closed);
92
93 m_output << " " << name << "=\"" << value << "\"";
94}
95
96/**
97 * \brief add an attribute to the current element
98 * \param name the attribute's name
99 * \param value the attribute's value
100 *
101 * This converts the integer to a string an uses the other overload.
102 */
103void XmlWriter::writeAttribute(const std::string& name, int value)
104{
105 writeAttribute(name, std::to_string(value));
106}

108/**
109 * \brief writes characters
110 * \param text the text to write
111 *
112 * This function inserts text under the current element (if any).
113 * Special XML characters, like '<' are escaped.
114 *
115 * When writing text inside an XML element, this function should only be called
116 * after all attributes have been written using writeAttribute().
117 */
118void XmlWriter::writeCharacters(const std::string& text)
119{
120 writeCharacters(std::string_view(text));
121}
122
123/**
124 * \overload void XmlWriter::writeCharacters(const std::string& text)
125 */
126void XmlWriter::writeCharacters(std::string_view text)
127{
128 writeCharacters(text.data(), text.size());
129}
130
131inline void write_char_escaped(XmlWriter::OutputStream& stream, char c)
132{
133 // https://stackoverflow.com/questions/1091945/what-characters-do-i-need-to-escape-in-xml-documents
134
135
136 if (c == '<')
137 stream << "&lt;";
138 else if (c == '>')
139 stream << "&gt;";
140 else if (c == '&')
141 stream << "&amp;";
142 else
143 stream << c;
144}
145
146/**
147 * \overload void XmlWriter::writeCharacters(const std::string& text)
148 */
149void XmlWriter::writeCharacters(const char* str)
150{
151 if (withinElement() && !current().attributes_closed)
152 sealAttributes();
153
154 while(*str)
155 {
156 char c = *(str++);
157 write_char_escaped(m_output, c);
158 }
159}
160
161/**
162 * \overload void XmlWriter::writeCharacters(const std::string& text)
163 */
164void XmlWriter::writeCharacters(const char* str, size_t n)
165{
166 if (withinElement() && !current().attributes_closed)
167 sealAttributes();
168
169 while(n--)
170 {
171 char c = *(str++);
172 write_char_escaped(m_output, c);
173 }
174}
175
176bool XmlWriter::withinElement() const
177{
178 return !m_elements.empty();
179}
180
181XmlWriter::Element& XmlWriter::current()
182{
183 return m_elements.back();
184}
185
186void XmlWriter::sealAttributes()
187{
188 m_output << ">";
189
190 current().attributes_closed = true;
191}
