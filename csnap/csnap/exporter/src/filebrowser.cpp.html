17void FileBrowserGenerator::generatePage()
18{
19 page.write("<!DOCTYPE html>\n");
20
21 html::start(page);
22 {
23 html::head(page);
24 {
25 html::title(page);
26 page << sema.file->path;
27 html::endtitle(page);
28
29 html::link(page, {
30 { "id", "codestylesheet" },
31 { "rel", "stylesheet" },
32 { "type", "text/css" },
33 { "href", page.url().pathToRoot() + "syntax.qtcreator.css" }
34 });
35 html::endlink(page);
36
37 html::link(page, {
38 { "rel", "stylesheet" },
39 { "type", "text/css" },
40 { "href", page.url().pathToRoot() + "tooltip.css" }
41 });
42 html::endlink(page);
43
44 html::script(page, {
45 { "type", "text/javascript" }
46 });
47 page << "var csnapRootPath=\"" << page.url().pathToRoot() << "\";";
48 html::endscript(page);
49
50 html::script(page, {
51 { "type", "text/javascript" },
52 { "src", page.url().pathToRoot() + "codenav.js" }
53 });
54 page << " ";
55 html::endscript(page);
56 }
57 html::endhead(page);
58
59 html::body(page);
60 {
61 generate();
62 }
63 html::endbody(page);
64 }
65 html::end(page);
66}

68void FileBrowserGenerator::generate()
69{
70 html::div(page, { {"id", "content"} });
71 {
72 if (!sema.reverse_includes.empty())
73 {
74 html::h2(page);
75 page << "Include graph";
76 html::endh2(page);
77
78 html::p(page);
79 page << "This file is included by the following file(s):";
80 html::endp(page);
81
82 html::ul(page);
83 for (const Include& inc : sema.reverse_includes)
84 {
85 const File* other = files.get(inc.file_id);
86 assert(other);
87
88 html::li(page);
89 {
90 html::a(page);
91 html::attr(page, "href", page.links().linkTo(*other, inc.line));
92 page << other->path << ":" << inc.line;
93 html::enda(page);
94 }
95 html::endli(page);
96 }
97 html::endul(page);
98 }
99
100 html::h2(page);
101 page << "Code";
102 html::endh2(page);
103
104 html::table(page, { {"class", "code"} });
105 page.xml.writeAttribute("file-id", sema.file->id.value());
106 {
107 html::tbody(page);
108 writeCode();
109 html::endtbody(page);
110 }
111 html::endtable(page);
112 }
113 html::enddiv(page);
114}
115
116void FileBrowserGenerator::generate(const std::set<int>& lines)
117{
118 html::table(page, { {"class", "code"} });
119 {
120 html::tbody(page);
121
122 for (int l : lines)
123 {
124 writeLine(l - 1);
125 }
126
127 html::endtbody(page);
128 }
129
130 html::endtable(page);
131}
132
133void FileBrowserGenerator::writeLine(size_t i)
134{
135 html::tr(page);
136
137 {
138 html::th(page, { {"id", "L" + std::to_string(i + 1)} });
139 page << std::to_string(i + 1);
140 html::endth(page);
141 }
142
143 {
144 html::td(page);
145 writeLineSource((int)i + 1);
146 html::endtd(page);
147 }
148
149 html::endtr(page);
150}

152void FileBrowserGenerator::writeCode()
153{
154 const std::vector<std::string_view>& lines = content.lines;
155
156 for (size_t i(0); i < lines.size(); ++i)
157 {
158 writeLine(i);
159 }
160}
