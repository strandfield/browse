8#include <cassert>
9#include <iterator>

11/**
12 * \brief a smart forward iterator that knows when end has been reached
13 *
14 * This class simply stores an iterator and the end-iterator.
15 */
16template<typename It>
17class Iterator
18{
19private:
20 It m_iterator;
21 It m_end;
22
23public:
24
25 using iterator_type = It;
26 using value_type = typename It::value_type;
27
28 Iterator() = delete;
29 Iterator(const Iterator<It>&) = default;
30 ~Iterator() = default;
31
32 /**
33 * \brief builds a smart iterator from a current and an end iterator
34 * \param current the current iterator
35 * \param end the end iterator
36 *
37 * The \a end iterator should be reachable by moving the \a current iterator
38 * forward a finite number of times.
39 */
40 Iterator(It current, It end) :
41 m_iterator(current),
42 m_end(end)
43 {
44 assert(std::distance(current, end) >= 0);
45 }
46
47 /**
48 * \brief returns the current value of the iterator
49 */
50 const It& iterator() const
51 {
52 return m_iterator;
53 }
54
55 /**
56 * \brief returns the end iterator
57 */
58 const It& end() const
59 {
60 return m_end;
61 }
62
63 /**
64 * \brief returns whether the iterator is at the end
65 *
66 * If atend() is true, this iterator should not be dereferenced.
67 */
68 bool atend() const
69 {
70 return iterator() == end();
71 }
72
73 /**
74 * \brief pre-increment operator
75 *
76 * Advances the iterator by one, and returns a reference to the iterator.
77 */
78 Iterator<It>& operator++()
79 {
80 ++m_iterator;
81 return *this;
82 }
83
84 /**
85 * \brief post-increment operator
86 *
87 * Advances the iterator by one, and returns a copy of the iterator before the
88 * increment.
89 */
90 Iterator<It> operator++(int)
91 {
92 Iterator<It> copy{ *this };
93 ++(*this);
94 return copy;
95 }
96
97 /**
98 * \brief dereference the iterator
99 */
100 const value_type& operator*() const
101 {
102 return *m_iterator;
103 }
104
105 Iterator<It>& operator=(const Iterator<It>&) = default;
106};
107
108#endif // CSNAP_ITERATOR_H
