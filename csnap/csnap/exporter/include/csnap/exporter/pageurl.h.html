47/**
48 * \brief returns a relative path to the root
49 *
50 * Example:
51 * \code{.cpp}
52 * PageURL("mypage.html").pathToRoot() == "";
53 * PageURL("folder/nested/mypage.html").pathToRoot() == "../../";
54 * \endcode
55 */
56inline std::string PageURL::pathToRoot() const
57{
58 std::string p = path().generic_string();
59 size_t nbdir = std::count(p.begin(), p.end(), '/');
60
61 std::string r;
62 r.reserve(nbdir * 3);
63
64 while (nbdir--)
65 {
66 r.append("../");
67 }
68
69 return r;
70}
