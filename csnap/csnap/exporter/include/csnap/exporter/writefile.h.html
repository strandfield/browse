15/**
16 * \brief writes a file on disk
17 * \param p the path on disk
18 * \param content the binary content of the file
19 *
20 * If the file already exists, it is overwritten.
21 *
22 * This function creates missing directories if \a p points
23 to a file in a directory that does not yet exist.
24 */
25inline void write_file(const std::filesystem::path& p, const std::string& content)
26{
27 std::filesystem::path dir = std::filesystem::absolute(p).parent_path();
28
29 if (!std::filesystem::exists(dir))
30 std::filesystem::create_directories(dir);
31
32 std::ofstream file{ p, std::ios::binary | std::ios::trunc };
33 file.write(content.data(), content.size());
34}
