61void scan(std::vector<std::string> args)
62{
63 using namespace csnap;
64
65 Scanner scanner;
66 scanner.save_ast = save_ast(args);
67 do_try([&scanner, &args]() { scanner.nb_parsing_threads = threads(args); });
68
69 std::filesystem::path dbpath = output(args);
70 std::filesystem::path slnpath = input(args);
71 bool should_overwrite = overwrite(args);
72
73 if (!args.empty())
74 {
75 std::cerr << "unrecognized command line args: ";
76
77 std::for_each(args.begin(), args.end(), [](const std::string& a) {
78 std::cerr << a << " ";
79 });
80
81 std::cerr << std::endl;
82 std::exit(1);
83 }
84
85 if (std::filesystem::exists(dbpath))
86 {
87 if (should_overwrite)
88 {
89 std::filesystem::remove(dbpath);
90 }
91 else
92 {
93 std::cerr << "output file already exists" << std::endl;
94 std::exit(1);
95 }
96 }
97
98 scanner.initSnapshot(dbpath);
99 scanner.scanSln(slnpath);
100}
