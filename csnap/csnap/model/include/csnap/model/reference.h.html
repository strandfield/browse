13/**
14 * \brief stores information about a reference to a symbol
15 */
16struct SymbolReference
17{
18 /**
19 * \brief the id of the symbol that is referenced
20 */
21 int symbol_id;
22 int file_id;
23 int line;
24 int col;
25
26 /**
27 * \brief the id of the symbol that is referencing the referenced symbol
28 */
29 SymbolId parent_symbol_id;
30
31 /**
32 * \brief a combination of flags as described by the Flag enum
33 */
34 int flags = 0;
35
36 /**
37 * \brief flag values for a symbol reference
38 *
39 * This enum mirrors libclang's CXSymbolRole.
40 */
41 enum Flag
42 {
43 Declaration = 1 << 0,
44 Definition = 1 << 1,
45 Reference = 1 << 2,
46 Read = 1 << 3,
47 Write = 1 << 4,
48 Call = 1 << 5,
49 Dynamic = 1 << 6,
50 AddressOf = 1 << 7,
51 Implicit = 1 << 8
52 };
53};
54
55inline bool operator==(const SymbolReference& lhs, const SymbolReference& rhs)
56{
57 return lhs.symbol_id == rhs.symbol_id
58 && lhs.file_id == rhs.file_id
59 && lhs.line == rhs.line
60 && lhs.col == rhs.col
61 && lhs.parent_symbol_id == rhs.parent_symbol_id
62 && lhs.flags == rhs.flags;
63}
64
65inline bool operator!=(const SymbolReference& lhs, const SymbolReference& rhs)
66{
67 return !(lhs == rhs);
68}
