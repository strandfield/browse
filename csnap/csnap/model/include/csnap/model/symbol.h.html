120inline AccessSpecifier get_access_specifier(const Symbol& s)
121{
122 switch (s.flags & Symbol::Private)
123 {
124 case Symbol::Public:
125 return AccessSpecifier::Public;
126 case Symbol::Protected:
127 return AccessSpecifier::Protected;
128 case Symbol::Private:
129 return AccessSpecifier::Private;
130 default:
131 return AccessSpecifier::Invalid;
132 }
133}
134
135inline void set_access_specifier(Symbol& s, AccessSpecifier a)
136{
137 s.flags = s.flags & ~Symbol::Private;
138
139 switch (a)
140 {
141 case AccessSpecifier::Public:
142 s.flags |= Symbol::Public;
143 break;
144 case AccessSpecifier::Protected:
145 s.flags |= Symbol::Protected;
146 break;
147 case AccessSpecifier::Private:
148 s.flags |= Symbol::Private;
149 break;
150 }
151}
