55inline void FileContent::splitContentIntoLines()
56{
57 lines.reserve(content.size() / 60);
58
59 size_t start = 0;
60
61 while (start < content.size())
62 {
63 size_t end = content.find('\n', start);
64
65 if (end == std::string::npos)
66 end = content.size();
67
68 auto line = std::string_view(content.data() + start, end - start);
69 lines.push_back(line);
70
71 start = end + 1;
72 }
73}
