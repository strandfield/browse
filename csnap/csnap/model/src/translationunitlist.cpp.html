12void TranslationUnitList::add(std::unique_ptr<TranslationUnit> tu)
13{
14 if (!tu)
15 return;
16
17 if (!tu->id.valid())
18 {
19 tu->id = TranslationUnitId((int)m_list.size());
20 }
21
22 size_t i = (size_t)tu->id.value();
23
24 if (i >= m_list.size())
25 m_list.resize(i + 1);
26
27 m_list[i] = std::move(tu);
28}
29
30/**
31 * \brief returns the list of all translation units as a vector
32 */
33std::vector<TranslationUnit*> TranslationUnitList::all() const
34{
35 std::vector<TranslationUnit*> r;
36 r.reserve(m_list.size());
37
38 std::for_each(m_list.begin(), m_list.end(), [&r](const TranslationUnitPtr& e) {
39 if (e) r.push_back(e.get());
40 });
41
42 return r;
43}
44
45/**
46 * \brief get a translation unit by id
47 * \param id the id of the translation unit
48 */
49TranslationUnit* TranslationUnitList::get(Identifier<TranslationUnit> id) const
50{
51 if (!id.valid())
52 return nullptr;
53
54 size_t offset = (size_t)id.value();
55
56 if (offset >= m_list.size())
57 return nullptr;
58
59 return m_list.at(offset).get();
60}
61
62/**
63 * \brief find a translation unit given a source file id
64 * \param id the id of the source file
65 */
66TranslationUnit* TranslationUnitList::find(FileId fid) const
67{
68 auto it = std::find_if(m_list.begin(), m_list.end(), [&fid](const TranslationUnitPtr& tu) {
69 return tu && tu->sourcefile_id == fid;
70 });
71
72 return it != m_list.end() ? it->get() : nullptr;
73}
74
75/**
76 * \brief returns the number of translation units in the list
77 */
78size_t TranslationUnitList::count() const
79{
80 return std::count_if(m_list.begin(), m_list.end(), [](const TranslationUnitPtr& tu) {
81 return tu != nullptr;
82 });
83}
