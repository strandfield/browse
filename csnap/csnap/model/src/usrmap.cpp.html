10/**
11 * \brief returns the id associated with a usr
12 * \param usr
13 *
14 * If the \a usr cannot be found in the map, this returns an invalid SymbolId.
15 */
16SymbolId UsrMap::find(const std::string& usr) const
17{
18 auto it = m_map.find(usr);
19 return it != m_map.end() ? it->second : SymbolId();
20}
21
22/**
23 * \brief inserts a new usr into the map
24 * \param usr
25 *
26 * This function assigns an unused id to \a usr and returns the id.
27 */
28SymbolId UsrMap::insert(const std::string& usr)
29{
30 SymbolId new_id{ (int)m_map.size() };
31 insert(usr, new_id);
32 return new_id;
33}
34
35/**
36 * \brief inserts a new usr into the map
37 * \param usr the usr
38 * \param id the id to assign to the usr
39 *
40 * Note that this function currently does not verify whether \a usr is already
41 * in the map or if \a id is already associated with another usr.
42 */
43void UsrMap::insert(const std::string& usr, SymbolId id)
44{
45 m_map[usr] = id;
46}
47
48/**
49 * \brief returns a copy of this usr map
50 */
51UsrMap GlobalUsrMap::clone()
52{
53 UsrMap copy;
54
55 {
56 std::lock_guard lock{ mutex() };
57 copy = usrMap();
58 }
59
60 return copy;
61}
62
63/**
64 * \brief get a symbol id from a usr
65 * \param usr the usr
66 *
67 * This function returns a pair with the symbol id and a boolean indicating
68 * whether the usr was inserted into the map (true if inserted; false if already present).
69 */
70std::pair<SymbolId, bool> GlobalUsrMap::get(const std::string& usr)
71{
72 std::lock_guard lock{ mutex() };
73
74 SymbolId id = usrMap().find(usr);
75
76 if (id.valid())
77 return { id, false };
78
79 return { usrMap().insert(usr), true };
80}

82UsrMap& GlobalUsrMap::usrMap()

84 return m_map;
85}
