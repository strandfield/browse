5#include "filecontentcache.h"
6
7#include <stdexcept>
8#include <vector>

10namespace csnap
11{
12
13void FileContentCache::insert(std::shared_ptr<FileContent> item)
14{
15 m_data[item->file->id.value()] = item;
16}
17
18std::shared_ptr<FileContent> FileContentCache::find(FileId itemid) const
19{
20 auto it = m_data.find(itemid.value());
21
22 if (it != m_data.end())
23 return it->second.lock();
24 else
25 return nullptr;
26}
27
28void FileContentCache::clear()
29{
30 m_data.clear();
31}
32
33void FileContentCache::cleanup()
34{
35 for (auto it = m_data.begin(); it != m_data.end(); )
36 {
37 if (it->second.expired())
38 it = m_data.erase(it);
39 else
40 ++it;
41 }
42}
