13File* FileList::add(std::string path)
14{
15 m_files.push_back(std::make_unique<File>());
16 m_files.back()->id = FileId((int)m_files.size() - 1);
17 m_files.back()->path = std::move(path);
18 return m_files.back().get();
19}
20
21File* FileList::add(std::unique_ptr<File> f)
22{
23 if(!f->id.valid())
24 f->id = FileId((int)m_files.size());
25
26 size_t i = (size_t)f->id.value();
27
28 if (i < m_files.size())
29 {
30 assert(m_files.at(i) == nullptr);
31 }
32
33 if (m_files.size() <= i)
34 {
35 m_files.resize(i + 1);
36 }
37
38 m_files[i] = std::move(f);
39
40 return m_files[i].get();
41}
42
43std::vector<File*> FileList::all() const
44{
45 std::vector<File*> r;
46 r.reserve(m_files.size());
47
48 std::for_each(m_files.begin(), m_files.end(), [&r](const std::unique_ptr<File>& f) {
49 if (f) r.push_back(f.get());
50 });
51
52 return r;
53}
54
55File* FileList::get(Identifier<File> id) const
56{
57 if (!id.valid())
58 return nullptr;
59
60 size_t offset = (size_t)id.value();
61
62 if (offset >= m_files.size())
63 return nullptr;
64
65 return m_files.at(offset).get();
66}
67
68File* FileList::find(const std::string& path) const
69{
70 auto it = std::find_if(m_files.begin(), m_files.end(), [&path](const FilePtr& f) {
71 return f && f->path == path;
72 });
73
74 return it != m_files.end() ? it->get() : nullptr;
75}
