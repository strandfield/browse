7#include <stdexcept>
8#include <vector>

10namespace csnap
11{
12
13void SymbolCache::insert(std::shared_ptr<Symbol> s)
14{
15 m_data[s->id.value()] = s;
16}
17
18std::shared_ptr<Symbol> SymbolCache::find(SymbolId symid) const
19{
20 auto it = m_data.find(symid.value());
21
22 if (it != m_data.end())
23 return it->second.lock();
24 else
25 return nullptr;
26}
27
28void SymbolCache::clear()
29{
30 m_data.clear();
31}
32
33void SymbolCache::cleanup()
34{
35 for (auto it = m_data.begin(); it != m_data.end(); )
36 {
37 if (it->second.expired())
38 it = m_data.erase(it);
39 else
40 ++it;
41 }
42}
