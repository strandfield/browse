7namespace csnap

9
10namespace version

12
13int major()

15 return CSNAP_VERSION_MAJOR;
16}
17
18int minor()

20 return CSNAP_VERSION_MINOR;
21}
22
23int patch()
24{

26}
27
28std::string suffix()
29{
30 return CSNAP_VERSION_SUFFIX;
31}
32
33} // namespace version
34
35/**
36 * \brief returns the csnap version as a string
37 */
38std::string versionstring()
39{
40 std::string r = std::to_string(version::major())
41 + "." + std::to_string(version::minor())
42 + "." + std::to_string(version::patch());
43
44 if (!version::suffix().empty())
45 r += "-" + version::suffix();
46
47 return r;
48}
