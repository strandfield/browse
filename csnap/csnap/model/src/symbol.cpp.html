13static std::string G_EMPTY_STR = "";
14
15static std::string G_STR_PUBLIC = "public";
16static std::string G_STR_PROTECTED = "protected";
17static std::string G_STR_PRIVATE = "private";
18
19const std::string& to_string(AccessSpecifier aspec)
20{
21 switch (aspec)
22 {
23 case AccessSpecifier::Public:
24 return G_STR_PUBLIC;
25 case AccessSpecifier::Protected:
26 return G_STR_PROTECTED;
27 case AccessSpecifier::Private:
28 return G_STR_PRIVATE;
29 default:
30 return G_EMPTY_STR;
31 }
32}
33
34const std::string& whatsit2string(Whatsit w)
35{
36
37 static const std::vector<std::string> strs = {
38 "Unexposed",
39 "Typedef",
40 "Function",
41 "Variable",
42 "Field",
43 "EnumConstant",
44 "ObjCClass",
45 "ObjCProtocol",
46 "ObjCCategory",
47 "ObjCInstanceMethod",
48 "ObjCClassMethod",
49 "ObjCProperty",
50 "ObjCIvar",
51 "Enum",
52 "Struct",
53 "Union",
54 "CXXClass",
55 "CXXNamespace",
56 "CXXNamespaceAlias",
57 "CXXStaticVariable",
58 "CXXStaticMethod",
59 "CXXInstanceMethod",
60 "CXXConstructor",
61 "CXXDestructor",
62 "CXXConversionFunction",
63 "CXXTypeAlias",
64 "CXXInterface",
65 };
66
67 auto i = static_cast<size_t>(w);
68
69 if (i >= strs.size())
70 return G_EMPTY_STR;
71 else
72 return strs.at(i);
73}
