28/**
29 * \brief loads a symbol from the database
30 * \param id the id of the symbol
31 * \return true on success, false otherwise
32 *
33 * This function can fail if @a id is invalid or if there is
34 * no symbol with the given @a id.
35 */
36bool SymbolLoader::read(SymbolId id)
37{
38 m_query.reset();
39
40 m_query.bind(1, id.value());
41
42 if (!m_query.step())
43 return false;
44
45 symbol.id = id;
46 symbol.kind = static_cast<Whatsit>(m_query.columnInt(1));
47
48 symbol.parent_id = m_query.nullColumn(2) ? SymbolId() : SymbolId(m_query.columnInt(2));
49 symbol.name = m_query.column(3);
50 symbol.usr = m_query.column(4);
51 symbol.display_name = m_query.nullColumn(5) ? std::string() : m_query.column(5);
52 symbol.flags = m_query.columnInt(6);
53
54 return true;
55}

74/**
75 * \brief load the next symbol
76 * \return true if a symbol was loaded, false if the end has been reached
77 */
78bool SymbolEnumerator::next()
79{
80 if (!m_query.step())
81 return false;
82
83 symbol.id = SymbolId(m_query.columnInt(0));
84 symbol.kind = static_cast<Whatsit>(m_query.columnInt(1));
85
86 symbol.parent_id = m_query.nullColumn(2) ? SymbolId() : SymbolId(m_query.columnInt(2));
87 symbol.name = m_query.column(3);
88 symbol.usr = m_query.column(4);
89 symbol.display_name = m_query.nullColumn(5) ? std::string() : m_query.column(5);
90 symbol.flags = m_query.columnInt(6);
91
92 return true;
93}
