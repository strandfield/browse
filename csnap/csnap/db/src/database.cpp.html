13namespace csnap

15
16Database::Database(Database&& other) :
17 m_database(other.m_database)
18{
19 other.m_database = nullptr;
20}
21
22Database::~Database()
23{
24 if (sqliteHandle())
25 close();
26}
27
28/**
29 * \brief return the underlying sqlite connection
30 */
31sqlite3* Database::sqliteHandle() const
32{
33 return m_database;
34}
35
36/**
37 * \brief returns whether an actual database connection is opened
38 */
39bool Database::good() const
40{
41 return sqliteHandle() != nullptr;
42}
43
44/**
45 * \brief opens a connection to a database
46 * \param dbPath the path of the database
47 * \return true on success, false otherwise
48 */
49bool Database::open(const std::filesystem::path& dbPath)
50{
51 int r = sqlite3_open(dbPath.u8string().c_str(), &m_database);
52 return r == SQLITE_OK;
53}
54
55/**
56 * \brief create a database and open a connection
57 * \param dbPath the path of the database
58 */
59void Database::create(const std::filesystem::path& dbPath)
60{
61 if (std::filesystem::exists(dbPath))
62 return;
63
64 int r = sqlite3_open_v2(dbPath.u8string().c_str(), &m_database, SQLITE_OPEN_READWRITE | SQLITE_OPEN_CREATE, NULL);
65
66 assert(r == SQLITE_OK);
67
68 if (r != SQLITE_OK)
69 {
70 std::cerr << "Failed to create database: error " << SQLITE_OK << std::endl;
71 }
72}
73
74/**
75 * \brief close the connection, if any
76 */
77void Database::close()
78{
79 if (!good())
80 return;
81
82 sqlite3_close(m_database);
83 m_database = nullptr;
84}
85
86Database& Database::operator=(Database&& other)
87{
88 m_database = other.m_database;
89 other.m_database = nullptr;
90 return *this;
91}
