150inline bool exec(Database& db, const std::string& query, std::string* error = nullptr)
151{
152 char* errorbuffer = nullptr;
153 int r = sqlite3_exec(db.sqliteHandle(), query.c_str(), NULL, NULL, &errorbuffer);
154
155 if (r != SQLITE_OK && error)
156 {
157 *error = errorbuffer;
158 sqlite3_free(errorbuffer);
159 }
160
161 return r == SQLITE_OK;
162}
